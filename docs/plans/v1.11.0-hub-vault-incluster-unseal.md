## Problem

The Tier 3 plan relocates the hub Vault into the Hostinger app cluster as the primary
(`HUB_VAULT_PROFILE=hostinger`). A bare Hostinger VPS has **no macOS Keychain and no managed
KMS**, so the laptop's unseal mechanism (Keychain-cached Shamir shards replayed by the Mac) does
not port. An in-cluster Vault that seals on every pod restart/reboot and never auto-unseals is
not a viable "primary."

**Chosen mechanism (locked 2026-06-27):** Shamir shards stored in the in-cluster `vault-unseal`
Secret (already created by `_vault_process_init_artifacts`), replayed by an **in-cluster
watchdog CronJob** that detects sealed state and unseals. No KMS, no Keychain, no laptop
dependency. (Init uses `-key-shares=1 -key-threshold=1`, so there is exactly one shard:
`shard-1`.)

This phase (P2a) ships **only the watchdog** — a self-contained, idempotent, safe-anywhere
building block. It does **not** provision the Hostinger Vault and does **not** flip any default;
applying the watchdog where no Vault/`vault-unseal` Secret exists is a graceful no-op (optional
secret mount + early exit). Behavior is preserved.

### Change 2 — NEW file `scripts/etc/vault/unseal-watchdog.yaml.tmpl`

Create this file exactly:

```yaml
apiVersion: batch/v1
kind: CronJob
metadata:
  name: vault-unseal-watchdog
  namespace: ${VAULT_NS}
  labels:
    app.kubernetes.io/name: vault-unseal-watchdog
    app.kubernetes.io/part-of: vault
spec:
  schedule: "* * * * *"
  concurrencyPolicy: Forbid
  successfulJobsHistoryLimit: 1
  failedJobsHistoryLimit: 3
  jobTemplate:
    spec:
      backoffLimit: 0
      activeDeadlineSeconds: 50
      template:
        metadata:
          labels:
            app.kubernetes.io/name: vault-unseal-watchdog
        spec:
          restartPolicy: Never
          securityContext:
            runAsNonRoot: true
            runAsUser: 100
          containers:
            - name: unseal
              image: ${VAULT_UNSEAL_IMAGE}
              imagePullPolicy: IfNotPresent
              env:
                - name: VAULT_ADDR
                  value: "${VAULT_ENDPOINT}"
              command:
                - /bin/sh
                - -c
                - |
                  set -u
                  SHARD_FILE=/etc/vault-unseal/shard-1
                  if [ ! -f "$SHARD_FILE" ]; then
                    echo "[unseal-watchdog] vault-unseal secret not mounted — nothing to do"
                    exit 0
                  fi
                  vault status >/dev/null 2>&1
                  rc=$?
                  if [ "$rc" -eq 0 ]; then
                    echo "[unseal-watchdog] vault already unsealed"
                    exit 0
                  fi
                  if [ "$rc" -eq 1 ]; then
                    echo "[unseal-watchdog] vault unreachable at $VAULT_ADDR — skipping"
                    exit 0
                  fi
                  echo "[unseal-watchdog] vault sealed — replaying shard"
                  vault operator unseal "$(cat "$SHARD_FILE")" >/dev/null 2>&1 || true
                  if vault status >/dev/null 2>&1; then
                    echo "[unseal-watchdog] unseal succeeded"
                  else
                    echo "[unseal-watchdog] unseal attempt complete (still sealed or unreachable)"
                  fi
              volumeMounts:
                - name: vault-unseal
                  mountPath: /etc/vault-unseal
                  readOnly: true
          volumes:
            - name: vault-unseal
              secret:
                secretName: vault-unseal
                optional: true
                items:
                  - key: shard-1
                    path: shard-1
```

> Notes for the implementer (do not add as comments to the file):
> - `vault status` exit codes: `0` = unsealed, `1` = error/unreachable, `2` = sealed.
> - The secret volume is `optional: true` and the script early-exits if `shard-1` is absent, so
>   the CronJob is a safe no-op until a Vault + `vault-unseal` Secret exist in the namespace.

# vault_install_unseal_watchdog [ns] [context]
# Renders + applies the in-cluster auto-unseal watchdog CronJob (Tier 3 P2a).
# Idempotent; safe to apply before a Vault exists (optional secret mount + early exit).
function vault_install_unseal_watchdog() {
   local ns="${1:-${VAULT_NS:-$VAULT_NS_DEFAULT}}"
   local app_context="${2:-}"
   _vault_source_optional_vars
   local template="$SCRIPT_DIR/etc/vault/unseal-watchdog.yaml.tmpl"
   [[ -f "$template" ]] || _err "[vault] missing template: $template"
   local rendered
   rendered="$(mktemp -t vault-unseal-watchdog.XXXXXX.yaml)"
   trap '$(_cleanup_trap_command "$rendered")' EXIT TERM
   VAULT_NS="$ns" \
   VAULT_UNSEAL_IMAGE="${VAULT_UNSEAL_IMAGE:-hashicorp/vault:1.18.3}" \
   VAULT_ENDPOINT="${VAULT_ENDPOINT:-http://vault.${ns}.svc:8200}" \
   envsubst '$VAULT_NS $VAULT_UNSEAL_IMAGE $VAULT_ENDPOINT' < "$template" > "$rendered"
   if [[ -n "$app_context" ]]; then
      kubectl apply --context "$app_context" -f "$rendered"
   else
      _kubectl apply -f "$rendered"
   fi
   _cleanup_on_success "$rendered"
   trap - EXIT TERM
}

function deploy_vault() {
```

> Do NOT wire `vault_install_unseal_watchdog` into any auto-run path. It is an operator command
> (dispatchable: `./scripts/k3d-manager vault_install_unseal_watchdog secrets <ctx>`) and the
> future P2b provisioning will call it. Leaving it un-called is intentional in P2a.

### Change 4 — NEW file `scripts/tests/etc/vault_unseal_watchdog.bats`

Pure logic (renders the template via `envsubst`, asserts the manifest). No cluster mocks.

```bash
#!/usr/bin/env bats

setup() {
  TMPL="${BATS_TEST_DIRNAME}/../../etc/vault/unseal-watchdog.yaml.tmpl"
  RENDERED="$(mktemp)"
  VAULT_NS="secrets" \
  VAULT_UNSEAL_IMAGE="hashicorp/vault:1.18.3" \
  VAULT_ENDPOINT="http://vault.secrets.svc:8200" \
  envsubst '$VAULT_NS $VAULT_UNSEAL_IMAGE $VAULT_ENDPOINT' < "$TMPL" > "$RENDERED"
}

teardown() {
  rm -f "$RENDERED"
}

@test "watchdog renders into the provided namespace" {
  grep -q "namespace: secrets" "$RENDERED"
}

@test "watchdog image is pinned (not latest)" {
  grep -q "image: hashicorp/vault:1.18.3" "$RENDERED"
  ! grep -qE "image:.*:latest" "$RENDERED"
}

@test "watchdog mounts the vault-unseal secret optionally" {
  grep -q "secretName: vault-unseal" "$RENDERED"
  grep -q "optional: true" "$RENDERED"
}

@test "watchdog targets the in-cluster endpoint and unseals via shard-1" {
  grep -q "http://vault.secrets.svc:8200" "$RENDERED"
  grep -q "/etc/vault-unseal/shard-1" "$RENDERED"
  grep -q "vault operator unseal" "$RENDERED"
}

@test "inline shell vars survived envsubst (not expanded away)" {
  grep -q 'rc=$?' "$RENDERED"
}
```
