## Problem

Both remote k3s providers (`k3s-aws`, `k3s-gcp`) install k3s with `--disable servicelb`
but never install a replacement load balancer controller or Istio. Result:

- `istio-ingressgateway` service stays `<pending>` — no external traffic into the cluster
- All Istio-dependent services (ArgoCD UI via VirtualService, Jenkins, Vault via Ingress) are
  unreachable from outside the node
- No mTLS between workloads

**k3s-aws (3-node, EC2 VPC):**
- MetalLB L2 mode works — EC2 instances in the same VPC subnet can ARP-reply for a VPC IP
- Use the server node's VPC internal IP (`InternalIP`) as the MetalLB address pool (/32)
- Ingressgateway gets that VPC IP; access is via the autossh tunnel (which already forwards
  to the VPC network)

**k3s-gcp (single-node, GCE):**
- GCE networking does not forward ARP for IPs not assigned to the NIC — MetalLB L2 won't work
- Use `externalIPs` on the ingressgateway Service set to the instance's GCE external IP
- Add GCP firewall rule for TCP 80/443/15021 (health check port for `istio-ingressgateway`)

---

## Fix

| Provider | LB strategy | Istio |
|----------|-------------|-------|
| `k3s-aws` | MetalLB L2, pool = server VPC internal IP /32 | install via existing template |
| `k3s-gcp` | `externalIPs` = GCE natIP + firewall rule | install via existing template |

## Change 1 — Create `scripts/plugins/service_mesh.sh`

```bash
#!/usr/bin/env bash
# scripts/plugins/service_mesh.sh — Istio + load balancer installation for remote k3s clusters

set -euo pipefail

function _service_mesh_install_istio() {
  local kubeconfig="$1"
  local istio_yaml_template="${SCRIPT_DIR}/etc/istio-operator.yaml.tmpl"
  local istio_var="${SCRIPT_DIR}/etc/istio_var.sh"

  [[ -r "${istio_yaml_template}" ]] || {
    _err "[service_mesh] Istio template not found: ${istio_yaml_template}"
    return 1
  }
  [[ -r "${istio_var}" ]] || {
    _err "[service_mesh] Istio vars not found: ${istio_var}"
    return 1
  }

  # shellcheck disable=SC1090
  source "${istio_var}"
  local istio_yamlfile
  istio_yamlfile=$(mktemp -t service-mesh-istio.XXXXXX.yaml)
  trap 'rm -f "${istio_yamlfile}"' EXIT

  envsubst < "${istio_yaml_template}" > "${istio_yamlfile}"

  _install_istioctl
  local -x KUBECONFIG="${kubeconfig}"
  _istioctl x precheck
  _istioctl install -y -f "${istio_yamlfile}"
  kubectl label ns default istio-injection=enabled --overwrite
  _info "[service_mesh] Istio installed."
}

function _service_mesh_apply_peer_authentication() {
  local kubeconfig="$1"
  _info "[service_mesh] Enforcing mTLS STRICT mode..."
  KUBECONFIG="${kubeconfig}" kubectl apply -f - <<'EOF'
apiVersion: security.istio.io/v1beta1
kind: PeerAuthentication
metadata:
  name: default
  namespace: istio-system
spec:
  mtls:
    mode: STRICT
EOF
  _info "[service_mesh] mTLS STRICT mode active."
}

function _service_mesh_install_metallb() {
  local kubeconfig="$1" pool_ip="$2"

  if ! [[ "${pool_ip}" =~ ^[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}$ ]]; then
    _err "[service_mesh] Invalid pool IP: ${pool_ip}"
    return 1
  fi

  _info "[service_mesh] Installing MetalLB (pool ${pool_ip}/32)..."
  local -x KUBECONFIG="${kubeconfig}"

  _helm repo add metallb https://metallb.github.io/metallb 2>/dev/null || true
  _helm repo update metallb
  _helm upgrade --install metallb metallb/metallb \
    --namespace metallb-system --create-namespace \
    --version 0.14.9 \
    --wait --timeout 180s

  _info "[service_mesh] Configuring IPAddressPool and L2Advertisement..."
  kubectl apply -f - <<EOF
apiVersion: metallb.io/v1beta1
kind: IPAddressPool
metadata:
  name: k3s-remote-pool
  namespace: metallb-system
spec:
  addresses:
  - ${pool_ip}/32
---
apiVersion: metallb.io/v1beta1
kind: L2Advertisement
metadata:
  name: k3s-remote-l2adv
  namespace: metallb-system
spec:
  ipAddressPools:
  - k3s-remote-pool
EOF
  _info "[service_mesh] MetalLB configured."
}

function _service_mesh_patch_gcp_external_ip() {
  local kubeconfig="$1" external_ip="$2"

  if ! [[ "${external_ip}" =~ ^[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}$ ]]; then
    _err "[service_mesh] Invalid external IP: ${external_ip}"
    return 1
  fi

  _info "[service_mesh] Patching istio-ingressgateway externalIPs → ${external_ip}..."
  KUBECONFIG="${kubeconfig}" kubectl patch service istio-ingressgateway \
    -n istio-system \
    --type='json' \
    -p="[{\"op\":\"add\",\"path\":\"/spec/externalIPs\",\"value\":[\"${external_ip}\"]}]"
  _info "[service_mesh] externalIPs set."
}

function deploy_service_mesh_aws() {
  local kubeconfig="${UBUNTU_K3S_LOCAL_KUBECONFIG:-${HOME}/.kube/k3s-ubuntu.yaml}"

  _info "[service_mesh] Getting server node VPC internal IP..."
  local server_ip
  server_ip=$(KUBECONFIG="${kubeconfig}" kubectl get nodes \
    --selector='node-role.kubernetes.io/control-plane' \
    -o jsonpath='{.items[0].status.addresses[?(@.type=="InternalIP")].address}' 2>/dev/null)

  if [[ -z "${server_ip}" ]]; then
    _err "[service_mesh] Could not determine server node InternalIP"
    return 1
  fi

  _service_mesh_install_istio "${kubeconfig}"
  _service_mesh_install_metallb "${kubeconfig}" "${server_ip}"
  _service_mesh_apply_peer_authentication "${kubeconfig}"

  _info "[service_mesh] k3s-aws: service mesh + MetalLB ready."
  _info "[service_mesh] IngressGateway IP: ${server_ip}"
  _info "[service_mesh] Verify: KUBECONFIG=${kubeconfig} kubectl -n istio-system get svc istio-ingressgateway"
}

function deploy_service_mesh_gcp() {
  local kubeconfig="${1:-${HOME}/.kube/k3s-gcp.yaml}"
  local external_ip="${2:-}"
  local project="${3:-${GCP_PROJECT:-}}"
  local network_tag="${4:-k3d-manager}"

  if [[ -z "${external_ip}" ]]; then
    _err "[service_mesh] deploy_service_mesh_gcp requires external_ip as \$2"
    return 1
  fi
  if [[ -z "${project}" ]]; then
    _err "[service_mesh] deploy_service_mesh_gcp requires project as \$3 or GCP_PROJECT env var"
    return 1
  fi

  _service_mesh_install_istio "${kubeconfig}"
  _service_mesh_patch_gcp_external_ip "${kubeconfig}" "${external_ip}"
  _service_mesh_apply_peer_authentication "${kubeconfig}"

  _info "[service_mesh] k3s-gcp: service mesh ready."
  _info "[service_mesh] IngressGateway external IP: ${external_ip}"
  _info "[service_mesh] Verify: KUBECONFIG=${kubeconfig} kubectl -n istio-system get svc istio-ingressgateway"
}
```

```bash
function _gcp_ensure_ingress_firewall() {
  local project="$1"
  local rule_name="k3d-manager-ingress"
  if gcloud compute firewall-rules describe "${rule_name}" \
      --project="${project}" --quiet >/dev/null 2>&1; then
    _info "[k3s-gcp] Firewall rule ${rule_name} already exists — skipping"
    return 0
  fi
  _info "[k3s-gcp] Creating firewall rule ${rule_name} (TCP 80/443/15021)..."
  gcloud compute firewall-rules create "${rule_name}" \
    --project="${project}" \
    --allow=tcp:80,tcp:443,tcp:15021 \
    --target-tags="${_GCP_NETWORK_TAG}" \
    --description="k3d-manager: Istio IngressGateway HTTP/HTTPS + health check" \
    --quiet
}

function _gcp_create_instance() {
```
