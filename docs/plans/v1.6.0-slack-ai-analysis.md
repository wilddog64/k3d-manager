# v1.6.0 — Slack slash commands, ACG expiry notification, ArgoCD alert + AI analysis

## Context

Four related capabilities sharing one Cloudflare Worker and the existing k3dm-webhook:

```
Slack slash command → Cloudflare Worker (validates sig, routes) → k3dm-webhook
                                                                       ↓ async job
                                                                   Slack response_url

k3dm-webhook _run_cluster (action=up) → writes acg-state ConfigMap on Hub

Hub CronJob (*/30 * * * *) → reads acg-state ConfigMap → Slack incoming webhook
  "ACG sandbox provisioned 3h 42m ago — may have expired. /acg-up to bring it back."

AlertManager → k3dm-webhook POST /api/v1/analyze
  → kubectl diagnostics → Claude API → Slack incoming webhook
```

**Slack slash commands supported:**

| Command | Action |
|---------|--------|
| `/acg-up [aws\|gcp]` | Bring up ACG cluster; default provider `aws` |
| `/acg-down` | Tear down ACG cluster (`KEEP_LOCAL=1` hardcoded) |
| `/acg-status` | Report node + ArgoCD app health; "cluster gone" if unreachable |
| `/argocd-upgrade <version> <stage>` | Trigger ArgoCD chart upgrade |

> **Note:** `azure` provider is on the roadmap (v1.4.0) but not yet in the Makefile.
> Worker accepts `azure` without error but k3dm-webhook maps unknown providers to `aws`.
> Do NOT add `k3s-azure` to the Makefile in this spec.

**ACG expiry:** Notification fires between 3.5h and 8h after last `make up` so you
can decide to `/acg-up`. No auto-action.

**ArgoCD alert:** `ArgoCDAppDegraded` PrometheusRule → AlertManager → `/api/v1/analyze`
→ Claude Haiku (`claude-haiku-4-5-20251001`) → Slack. Existing degraded apps
(`data-layer`, `shopping-cart-apps`) serve as live test targets — do NOT fix them.

**Prerequisites (already implemented):** `bin/k3dm-webhook` (v1.6.0 webhook specs done).

## File 1: `workers/slack-relay/index.js`

Create directory `workers/slack-relay/` and this file.

Validates Slack request signature (HMAC-SHA256), parses slash command, relays to
k3dm-webhook, returns immediate 200 to Slack. Job results delivered async via
`response_url` by k3dm-webhook.

```javascript
const ALLOWED_COMMANDS = new Set(['/acg-up', '/acg-down', '/acg-status', '/argocd-upgrade'])
const VALID_PROVIDERS   = new Set(['aws', 'gcp', 'azure'])

async function verifySlack(request, body) {
  const ts  = request.headers.get('X-Slack-Request-Timestamp') || ''
  const sig = request.headers.get('X-Slack-Signature') || ''
  if (!ts || !sig) return false
  if (Math.abs(Date.now() / 1000 - Number(ts)) > 300) return false

  const enc = new TextEncoder()
  const key = await crypto.subtle.importKey(
    'raw', enc.encode(SLACK_SIGNING_SECRET),
    { name: 'HMAC', hash: 'SHA-256' }, false, ['sign']
  )
  const raw = await crypto.subtle.sign('HMAC', key, enc.encode(`v0:${ts}:${body}`))
  const hex = Array.from(new Uint8Array(raw)).map(b => b.toString(16).padStart(2, '0')).join('')
  const expected = `v0=${hex}`
  if (expected.length !== sig.length) return false
  let diff = 0
  for (let i = 0; i < expected.length; i++) diff |= expected.charCodeAt(i) ^ sig.charCodeAt(i)
  return diff === 0
}

async function relay(endpoint, payload, event) {
  event.waitUntil(
    fetch(`${WEBHOOK_URL}${endpoint}`, {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${WEBHOOK_TOKEN}`,
        'Content-Type':  'application/json'
      },
      body: JSON.stringify(payload)
    })
  )
}

function jsonReply(text) {
  return new Response(JSON.stringify({ text, response_type: 'in_channel' }),
    { status: 200, headers: { 'Content-Type': 'application/json' } })
}

addEventListener('fetch', event => {
  event.respondWith(handle(event))
})

async function handle(event) {
  const req = event.request
  if (req.method !== 'POST') return new Response('Not Found', { status: 404 })

  const body = await req.text()
  if (!await verifySlack(req, body)) return new Response('Unauthorized', { status: 401 })

  const p           = new URLSearchParams(body)
  const command     = p.get('command')    || ''
  const text        = (p.get('text')      || '').trim()
  const responseUrl = p.get('response_url') || ''

  if (!ALLOWED_COMMANDS.has(command)) return jsonReply(`Unknown command: ${command}`)

  if (command === '/acg-up') {
    const provider = VALID_PROVIDERS.has(text) ? text : 'aws'
    await relay('/api/v1/cluster', { action: 'up', provider, response_url: responseUrl }, event)
    return jsonReply(`⏳ Bringing up ACG cluster (${provider})…`)
  }

  if (command === '/acg-down') {
    await relay('/api/v1/cluster', { action: 'down', response_url: responseUrl }, event)
    return jsonReply('⏳ Tearing down ACG cluster…')
  }

  if (command === '/acg-status') {
    await relay('/api/v1/cluster-status', { response_url: responseUrl }, event)
    return jsonReply('🔍 Checking ACG cluster status…')
  }

  if (command === '/argocd-upgrade') {
    const parts   = text.split(/\s+/)
    const version = parts[0] || ''
    const stage   = parts[1] || 'infra'
    if (!version) return jsonReply('Usage: /argocd-upgrade <chart_version> [acg|infra]')
    if (!['acg', 'infra'].includes(stage)) return jsonReply('stage must be acg or infra')
    await relay('/api/v1/argocd-upgrade',
      { chart_version: version, stage, response_url: responseUrl }, event)
    return jsonReply(`⏳ Upgrading ArgoCD to chart ${version} on ${stage}…`)
  }

  return jsonReply('Unhandled command')
}
```

```python
def _record_acg_state(provider):
    """Write provision timestamp to Hub platform-ops ConfigMap."""
    ts = str(int(datetime.datetime.utcnow().timestamp()))
    create = subprocess.run(
        ["kubectl", "create", "configmap", "acg-state",
         f"--from-literal=provisioned-at={ts}",
         f"--from-literal=provider={provider}",
         "-n", "platform-ops",
         "--dry-run=client", "-o", "yaml"],
        capture_output=True, text=True
    )
    if create.returncode == 0:
        subprocess.run(
            ["kubectl", "apply", "-f", "-"],
            input=create.stdout,
            capture_output=True
        )


def _slack_post(url, text):
    """POST a text message to a Slack incoming webhook or response_url."""
    if not url:
        return
    data = json.dumps({"text": text, "response_type": "in_channel"}).encode()
    req = urllib.request.Request(url, data=data,
                                 headers={"Content-Type": "application/json"})
    try:
        urllib.request.urlopen(req, timeout=10)
    except Exception:
        pass


def _call_claude(prompt):
    """Call Anthropic Messages API with stdlib urllib. Returns text or error string."""
    if not ANTHROPIC_API_KEY:
        return "ANTHROPIC_API_KEY not configured — skipping AI analysis"
    data = json.dumps({
        "model": ANALYSIS_MODEL,
        "max_tokens": 1024,
        "messages": [{"role": "user", "content": prompt}]
    }).encode()
    req = urllib.request.Request(
        "https://api.anthropic.com/v1/messages",
        data=data,
        headers={
            "x-api-key": ANTHROPIC_API_KEY,
            "anthropic-version": "2023-06-01",
            "content-type": "application/json"
        }
    )
    try:
        with urllib.request.urlopen(req, timeout=30) as resp:
            result = json.loads(resp.read())
            return result["content"][0]["text"]
    except Exception as exc:
        return f"Claude API error: {exc}"


def _run_cluster_status(job_id, response_url):
    """Check ACG cluster reachability + ArgoCD app health; post result to response_url."""
    job_dir = JOB_DIR / job_id
    lines = []

    def _log(msg):
        lines.append(msg + "\n")

    def _finish(status):
        (job_dir / "status").write_text(status)
        (job_dir / "output").write_text("".join(lines))
        if response_url:
            _slack_post(response_url, "".join(lines))

    try:
        (job_dir / "status").write_text("running")

        nodes = subprocess.run(
            ["kubectl", "get", "nodes", "--context", "ubuntu-k3s",
             "--request-timeout=8s", "--no-headers"],
            capture_output=True, text=True
        )
        if nodes.returncode != 0:
            _log("❌ *ACG cluster unreachable* — sandbox may have expired.")
            _finish("success")
            return

        _log("✅ *ACG cluster nodes:*")
        _log(f"```{nodes.stdout.strip()}```")

        apps = subprocess.run(
            ["kubectl", "get", "applications", "-n", "cicd",
             "--context", "ubuntu-k3s",
             "--no-headers",
             "-o", "custom-columns=NAME:.metadata.name,SYNC:.status.sync.status,HEALTH:.status.health.status"],
            capture_output=True, text=True
        )
        if apps.returncode == 0 and apps.stdout.strip():
            _log("\n*ArgoCD apps:*")
            _log(f"```{apps.stdout.strip()}```")
        else:
            _log("\n_ArgoCD app status unavailable_")

        _finish("success")
    except Exception as exc:
        _log(f"ERROR: {exc}")
        _finish("failed")


def _run_analyze(job_id, alert_payload, response_url=""):
    """Gather kubectl diagnostics for firing alerts; call Claude; post to Slack."""
    job_dir = JOB_DIR / job_id
    lines = []

    def _log(msg):
        lines.append(msg + "\n")

    def _finish(status):
        (job_dir / "status").write_text(status)
        (job_dir / "output").write_text("".join(lines))

    try:
        (job_dir / "status").write_text("running")
        alerts = alert_payload.get("alerts", [])
        firing = [a for a in alerts if a.get("status") == "firing"]
        if not firing:
            _finish("success")
            return

        for alert in firing:
            app_name = alert.get("labels", {}).get("name", "")
            alert_name = alert.get("labels", {}).get("alertname", "unknown")
            _log(f"=== {alert_name}: {app_name} ===\n")

            diag_parts = []
            if app_name:
                desc = subprocess.run(
                    ["kubectl", "describe", "application", app_name, "-n", "cicd"],
                    capture_output=True, text=True
                )
                if desc.returncode == 0:
                    diag_parts.append(f"ArgoCD Application:\n{desc.stdout[-3000:]}")

            pods = subprocess.run(
                ["kubectl", "get", "pods", "-A",
                 "--field-selector=status.phase!=Running",
                 "--no-headers"],
                capture_output=True, text=True
            )
            if pods.returncode == 0 and pods.stdout.strip():
                diag_parts.append(f"Non-running pods:\n{pods.stdout.strip()}")

            if not diag_parts:
                _log("No diagnostics gathered\n")
                continue

            diagnostics = "\n\n".join(diag_parts)
            prompt = (
                f"You are a Kubernetes operations expert. Analyze this alert and give a "
                f"concise diagnosis (2-3 sentences) and the single most likely fix.\n\n"
                f"Alert: {alert_name} on app '{app_name}'\n\n"
                f"Diagnostics:\n{diagnostics}"
            )
            analysis = _call_claude(prompt)
            _log(analysis + "\n")

            notify_url = response_url or SLACK_WEBHOOK_URL
            if notify_url:
                _slack_post(notify_url,
                    f"🔍 *{alert_name}* — `{app_name}`\n{analysis}")

        _finish("success")
    except Exception as exc:
        _log(f"ERROR: {exc}")
        _finish("failed")
```

## File 7: `scripts/etc/argocd/platform-ops/acg-expiry.sh`

Create with mode `0755`. POSIX sh, `set -eu`.

```sh
#!/bin/sh
# acg-expiry.sh — notify via Slack when ACG sandbox may have expired.
# Fires between ACG_EXPIRY_WARNING_SECONDS and ACG_EXPIRY_MAX_SECONDS after last provision.

set -eu

SLACK_URL="${K3DM_SLACK_WEBHOOK_URL:-}"
WARNING_S="${ACG_EXPIRY_WARNING_SECONDS:-12600}"
MAX_S="${ACG_EXPIRY_MAX_SECONDS:-28800}"
STATE_FILE="/config/acg-state/provisioned-at"
PROVIDER_FILE="/config/acg-state/provider"

if [ ! -f "${STATE_FILE}" ] || [ ! -s "${STATE_FILE}" ]; then
  exit 0
fi

PROVISIONED_AT=$(cat "${STATE_FILE}")
PROVIDER=$(cat "${PROVIDER_FILE}" 2>/dev/null || echo "aws")
NOW=$(date +%s)
AGE=$(( NOW - PROVISIONED_AT ))

if [ "${AGE}" -lt "${WARNING_S}" ]; then
  exit 0
fi

if [ "${AGE}" -gt "${MAX_S}" ]; then
  exit 0
fi

if [ -z "${SLACK_URL}" ]; then
  echo "[acg-expiry] Slack webhook URL not configured"
  exit 0
fi

AGE_H=$(( AGE / 3600 ))
AGE_M=$(( (AGE % 3600) / 60 ))

curl -sf --max-time 10 -X POST "${SLACK_URL}" \
```
