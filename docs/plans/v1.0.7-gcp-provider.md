# Spec: v1.0.7 — GCP Provider (`k3s-gcp`)

## Context

Add `CLUSTER_PROVIDER=k3s-gcp` to the provider system. Uses `gcloud` CLI imperative
calls (no Deployment Manager, no Terraform) — consistent with the AWS provider's pattern
of shell-wrapping a cloud CLI. Targets Pluralsight ACG GCP sandbox (free tier GCE
instances). No managed Kubernetes (GKE) — k3s on bare GCE VM via k3sup.

**Scope of this spec:** Provider skeleton + `gcloud`-based compute provisioning.
GCP credential extraction via Playwright is handled in a separate spec
(`v1.0.7-playwright-provider-flag.md`).

**Known unknown:** Pluralsight GCP sandbox constraints (allowed zones, machine types,
project ID format, service account permissions) are not confirmed until a sandbox is
inspected. Placeholder values below are derived from free-tier GCE defaults; update
before handing to Codex.

### New file: `scripts/lib/providers/k3s-gcp.sh`

```bash
# shellcheck shell=bash
# scripts/lib/providers/k3s-gcp.sh — k3s on Pluralsight ACG GCP sandbox (single-node)
#
# Provider actions:
#   deploy_cluster  — gcp_get_credentials → gcloud compute instance create
#                     → k3sup install → kubeconfig merge → kubectl label
#   destroy_cluster — gcloud compute instance delete
#
# Config (env overrides):
#   GCP_PROJECT       GCP project ID (from sandbox credentials)
#   GCP_ZONE          Compute zone (default: us-central1-a)
#   GCP_MACHINE_TYPE  Instance machine type (default: e2-medium)
#   GCP_INSTANCE_NAME VM instance name (default: k3s-gcp-server)
#   GCP_SSH_KEY_FILE  Path to SSH key for k3sup (default: ~/.ssh/k3d-manager-gcp-key)

# shellcheck source=/dev/null
source "${SCRIPT_DIR}/plugins/gcp.sh"

# ── constants ────────────────────────────────────────────────────────────────
_GCP_ZONE="${GCP_ZONE:-us-central1-a}"
_GCP_MACHINE_TYPE="${GCP_MACHINE_TYPE:-e2-medium}"
_GCP_INSTANCE_NAME="${GCP_INSTANCE_NAME:-k3s-gcp-server}"
_GCP_SSH_KEY_FILE="${GCP_SSH_KEY_FILE:-${HOME}/.ssh/k3d-manager-gcp-key}"
_GCP_KUBECONFIG="${HOME}/.kube/k3s-gcp.yaml"

function _provider_k3s_gcp_deploy_cluster() {
  if [[ "${1:-}" == "-h" || "${1:-}" == "--help" ]]; then
    cat <<'HELP'
Usage: CLUSTER_PROVIDER=k3s-gcp ./scripts/k3d-manager deploy_cluster

Provision a single-node k3s cluster on ACG GCP sandbox:
  1. gcp_get_credentials        — extract GCP project + auth from Pluralsight UI
  2. gcloud auth activate       — configure gcloud with sandbox credentials
  3. gcloud compute create      — provision GCE instance
  4. k3sup install              — install k3s on the instance via SSH
  5. kubeconfig merge           — merge ~/.kube/k3s-gcp.yaml into default kubeconfig
  6. kubectl label              — k3d-manager/node-type=server

Milestone gate: kubectl --context k3s-gcp get nodes shows 1 node Ready.

Config (env overrides):
  GCP_PROJECT       GCP project ID (from sandbox credentials)
  GCP_ZONE          Compute zone (default: us-central1-a)
  GCP_MACHINE_TYPE  Instance machine type (default: e2-medium)
  GCP_INSTANCE_NAME VM instance name (default: k3s-gcp-server)
  GCP_SSH_KEY_FILE  SSH key for k3sup (default: ~/.ssh/k3d-manager-gcp-key)
HELP
    return 0
  fi

  _info "[k3s-gcp] Extracting GCP sandbox credentials..."
  gcp_get_credentials "${_GCP_SANDBOX_URL:-}" || return 1

  local project="${GCP_PROJECT:-}"
  local key_file="${GOOGLE_APPLICATION_CREDENTIALS:-}"
  if [[ -z "${project}" ]]; then
    _err "[k3s-gcp] GCP_PROJECT is not set — credential extraction may have failed"
    return 1
  fi
  if [[ -z "${key_file}" || ! -f "${key_file}" ]]; then
    _err "[k3s-gcp] GOOGLE_APPLICATION_CREDENTIALS not set or key file not found: ${key_file}"
    return 1
  fi

  _info "[k3s-gcp] Activating service account for project ${project}..."
  gcloud auth activate-service-account --key-file="${key_file}" || return 1
  gcloud config set project "${project}" || return 1

  _info "[k3s-gcp] Checking for existing instance ${_GCP_INSTANCE_NAME}..."
  local existing
  existing=$(gcloud compute instances describe "${_GCP_INSTANCE_NAME}" \
    --zone="${_GCP_ZONE}" --format="value(name)" 2>/dev/null || true)
  if [[ -n "${existing}" ]]; then
    _info "[k3s-gcp] Instance ${_GCP_INSTANCE_NAME} already exists — skipping create"
  else
    _info "[k3s-gcp] Creating GCE instance ${_GCP_INSTANCE_NAME} (${_GCP_MACHINE_TYPE}, ${_GCP_ZONE})..."
    gcloud compute instances create "${_GCP_INSTANCE_NAME}" \
      --zone="${_GCP_ZONE}" \
      --machine-type="${_GCP_MACHINE_TYPE}" \
      --image-family=ubuntu-2204-lts \
      --image-project=ubuntu-os-cloud \
      --tags=k3s-server \
      --metadata="ssh-keys=ubuntu:$(cat "${_GCP_SSH_KEY_FILE}.pub")" \
      --quiet || return 1
  fi

  _info "[k3s-gcp] Waiting for instance to be RUNNING..."
  local attempts=0
  until gcloud compute instances describe "${_GCP_INSTANCE_NAME}" \
      --zone="${_GCP_ZONE}" --format="value(status)" 2>/dev/null | grep -q "RUNNING"; do
    (( attempts++ ))
    if (( attempts >= 24 )); then
      _err "[k3s-gcp] Instance did not reach RUNNING state after 120s"
      return 1
    fi
    sleep 5
  done

  local external_ip
  external_ip=$(gcloud compute instances describe "${_GCP_INSTANCE_NAME}" \
    --zone="${_GCP_ZONE}" \
    --format="value(networkInterfaces[0].accessConfigs[0].natIP)" 2>/dev/null)
  if [[ -z "${external_ip}" || "${external_ip}" == "None" || "${external_ip}" == "null" ]]; then
    _err "[k3s-gcp] Could not retrieve external IP for ${_GCP_INSTANCE_NAME}"
    return 1
  fi
  _info "[k3s-gcp] Instance IP: ${external_ip}"

  _info "[k3s-gcp] Installing k3s via k3sup..."
  _ensure_k3sup || return 1
  mkdir -p "${HOME}/.kube"
  k3sup install \
    --ip="${external_ip}" \
    --user=ubuntu \
    --ssh-key="${_GCP_SSH_KEY_FILE}" \
    --local-path="${_GCP_KUBECONFIG}" \
    --context=k3s-gcp \
    --k3s-extra-args="--disable=traefik" || return 1

  _info "[k3s-gcp] Merging kubeconfig..."
  local merged
  merged=$(KUBECONFIG="${_GCP_KUBECONFIG}:${HOME}/.kube/config" \
    kubectl config view --flatten 2>/dev/null)
  if [[ -n "${merged}" ]]; then
    printf '%s\n' "${merged}" > "${HOME}/.kube/config"
  fi

  _info "[k3s-gcp] Waiting for node Ready..."
  local node_attempts=0
  until [[ "$(KUBECONFIG="${_GCP_KUBECONFIG}" kubectl get nodes --no-headers 2>/dev/null | grep -c " Ready")" -ge 1 ]]; do
    (( node_attempts++ ))
    if (( node_attempts >= 36 )); then
      _info "[k3s-gcp] WARNING: node not Ready after 180s — skipping label step"
      break
    fi
    sleep 5
  done

  if (( node_attempts < 36 )); then
    _info "[k3s-gcp] Labeling server node..."
    local server_node
    server_node=$(KUBECONFIG="${_GCP_KUBECONFIG}" kubectl get nodes \
      --no-headers -o custom-columns=NAME:.metadata.name 2>/dev/null | head -n1)
    [[ -n "${server_node}" ]] && \
      KUBECONFIG="${_GCP_KUBECONFIG}" kubectl label node "${server_node}" \
        k3d-manager/node-type=server --overwrite >/dev/null 2>&1 || true
  fi

  _info "[k3s-gcp] Cluster ready. Single-node k3s on GCE provisioned."
  _info "[k3s-gcp] Verify: kubectl --context k3s-gcp get nodes"
}

function _provider_k3s_gcp_destroy_cluster() {
  if [[ "${1:-}" == "-h" || "${1:-}" == "--help" ]]; then
    cat <<'HELP'
Usage: CLUSTER_PROVIDER=k3s-gcp ./scripts/k3d-manager destroy_cluster --confirm

Tear down the k3s-gcp cluster:
  1. gcloud compute instances delete  — delete GCE instance
  2. Remove k3s-gcp kubeconfig context

Requires --confirm to prevent accidental teardown.
HELP
    return 0
  fi

  if [[ "${1:-}" != "--confirm" ]]; then
    printf 'ERROR: %s\n' "[k3s-gcp] destroy_cluster requires --confirm" >&2
    return 1
  fi

  _info "[k3s-gcp] Deleting GCE instance ${_GCP_INSTANCE_NAME}..."
  local existing
  existing=$(gcloud compute instances describe "${_GCP_INSTANCE_NAME}" \
    --zone="${_GCP_ZONE}" --format="value(name)" 2>/dev/null || true)
  if [[ -z "${existing}" ]]; then
    _info "[k3s-gcp] Instance ${_GCP_INSTANCE_NAME} not found — skipping delete"
  else
    gcloud compute instances delete "${_GCP_INSTANCE_NAME}" \
      --zone="${_GCP_ZONE}" --quiet || return 1
  fi

  _info "[k3s-gcp] Removing kubeconfig context k3s-gcp..."
  kubectl config delete-context k3s-gcp 2>/dev/null || true
  kubectl config delete-cluster k3s-gcp 2>/dev/null || true
  rm -f "${_GCP_KUBECONFIG}"

  _info "[k3s-gcp] Cluster destroyed."
}
```

## New file: `scripts/plugins/gcp.sh`

Credential extraction confirmed from Pluralsight GCP sandbox UI (2026-04-11 screenshot):
- Three `input[aria-label="Copyable input"]` fields: Username, Password, Service Account Credentials
- Service Account Credentials is a full JSON blob — `project_id` parsed from it
- `gcloud` auth path: `gcloud auth activate-service-account --key-file=<written JSON file>`
- Username/Password are for GCP Console web UI only — not used by `gcloud` CLI

```bash
#!/usr/bin/env bash
# scripts/plugins/gcp.sh — GCP sandbox credential management
#
# Public functions:
#   gcp_get_credentials <url>  — extract GCP credentials from Pluralsight UI via Playwright
#
# Exports (consumed by k3s-gcp provider):
#   GCP_PROJECT                    — GCP project ID (parsed from service account JSON)
#   GOOGLE_APPLICATION_CREDENTIALS — path to written service account key file
#   GCP_USERNAME                   — sandbox console login email (informational)
#   GCP_PASSWORD                   — sandbox console login password (informational)

set -euo pipefail

function gcp_get_credentials() {
  local url="${1:-${_GCP_SANDBOX_URL:-}}"
  if [[ -z "${url}" ]]; then
    _err "[gcp] No sandbox URL provided — set GCP_SANDBOX_URL or pass as argument"
    return 1
  fi

  _info "[gcp] Extracting GCP credentials from Pluralsight sandbox..."
  local output
  output=$(node "${SCRIPT_DIR}/playwright/acg_credentials.js" "${url}" --provider gcp 2>/tmp/gcp_creds_err) \
    || { cat /tmp/gcp_creds_err >&2; _err "[gcp] Playwright credential extraction failed"; return 1; }

  local project key_path
  project=$(printf '%s\n' "${output}" | grep '^GCP_PROJECT=' | cut -d= -f2-)
  key_path=$(printf '%s\n' "${output}" | grep '^GOOGLE_APPLICATION_CREDENTIALS=' | cut -d= -f2-)

  if [[ -z "${project}" || "${project}" == "None" || "${project}" == "null" ]]; then
    _err "[gcp] Could not extract GCP_PROJECT from Playwright output"
    return 1
  fi
  if [[ -z "${key_path}" || ! -f "${key_path}" ]]; then
    _err "[gcp] GOOGLE_APPLICATION_CREDENTIALS not set or key file not found: ${key_path}"
    return 1
  fi

  export GCP_PROJECT="${project}"
  export GOOGLE_APPLICATION_CREDENTIALS="${key_path}"
  export GCP_USERNAME
  GCP_USERNAME=$(printf '%s\n' "${output}" | grep '^GCP_USERNAME=' | cut -d= -f2-)
  export GCP_PASSWORD
  GCP_PASSWORD=$(printf '%s\n' "${output}" | grep '^GCP_PASSWORD=' | cut -d= -f2-)

  _info "[gcp] GCP_PROJECT=${project}"
  _info "[gcp] GOOGLE_APPLICATION_CREDENTIALS=${key_path}"
}
```

## SSH Key Pre-requisite

Before `deploy_cluster` is called, `_GCP_SSH_KEY_FILE` must exist. Provider does not
generate SSH keys — user must have run:

```bash
ssh-keygen -t ed25519 -f ~/.ssh/k3d-manager-gcp-key -N ""
```

Add a guard at the top of `_provider_k3s_gcp_deploy_cluster`:

```bash
if [[ ! -f "${_GCP_SSH_KEY_FILE}" || ! -f "${_GCP_SSH_KEY_FILE}.pub" ]]; then
  _err "[k3s-gcp] SSH key not found: ${_GCP_SSH_KEY_FILE}"
  _err "[k3s-gcp] Generate with: ssh-keygen -t ed25519 -f ${_GCP_SSH_KEY_FILE} -N \"\""
  return 1
fi
```
