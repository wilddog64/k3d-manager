## Step 1 — Diagnose: dump all buttons on the sandbox page

Run this one-off Node.js script to capture every visible button on the page:

```javascript
// scripts/playwright/debug_buttons.js — run once, do not commit
const { chromium } = require('playwright');
const os = require('os');
const path = require('path');

const AUTH_DIR = path.join(os.homedir(), '.local', 'share', 'k3d-manager', 'playwright-auth');
const URL = process.argv[2] || 'https://app.pluralsight.com/cloud-playground/cloud-sandboxes';

(async () => {
  const ctx = await chromium.launchPersistentContext(AUTH_DIR, {
    headless: false, channel: 'chrome', args: ['--password-store=basic'],
  });
  const page = ctx.pages()[0] || await ctx.newPage();
  await page.goto(URL, { waitUntil: 'domcontentloaded', timeout: 60000 });
  await page.waitForTimeout(5000);

  const buttons = await page.$$eval('button, [role="button"], a[class*="button"]', els =>
    els.map(el => ({
      text: el.innerText?.trim().slice(0, 80),
      ariaLabel: el.getAttribute('aria-label'),
      dataTestId: el.getAttribute('data-testid'),
      className: el.className?.slice(0, 60),
      visible: el.offsetParent !== null,
    }))
  );
  console.log(JSON.stringify(buttons, null, 2));
  await ctx.close();
})();
```

Run: `node scripts/playwright/debug_buttons.js "https://app.pluralsight.com/cloud-playground/cloud-sandboxes" 2>/dev/null`

Paste the output — look for buttons with text or aria-label containing "extend", "renew", "+4", "add time", or similar.
