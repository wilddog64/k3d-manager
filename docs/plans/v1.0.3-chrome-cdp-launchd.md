**Date**: 2026-04-04
**Branch**: k3d-manager-v1.0.3
**Status**: ASSIGNED to Codex

## Problem

`acg_credentials.js` uses CDP (`connectOverCDP('http://localhost:9222')`) to reuse the user's
live Pluralsight session. This only works if Chrome was launched with `--remote-debugging-port=9222`.
The user's regular Chrome (opened normally) does not have CDP enabled, so the probe always fails
and the script falls back to `launchPersistentContext` ‚Äî which launches a second Chrome window and
requires a fresh Pluralsight login when session cookies expire (e.g. overnight).

Fix: install a launchd agent on macOS that starts a dedicated Chrome instance with CDP on port 9222,
using the Playwright auth dir as its profile. This instance stays running across sessions, keeps
Pluralsight cookies fresh, and is always reachable via CDP.

### Add private helper `_acg_chrome_cdp_write_plist` (after `_acg_watch_write_plist` function)

```bash
_acg_chrome_cdp_write_plist() {
  mkdir -p "$(dirname "${_ACG_CHROME_CDP_PLIST}")"
  mkdir -p "${_ACG_CHROME_CDP_AUTH_DIR}"
  cat > "${_ACG_CHROME_CDP_PLIST}" <<PLIST
<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE plist PUBLIC "-//Apple//DTD PLIST 1.0//EN" "http://www.apple.com/DTDs/PropertyList-1.0.dtd">
<plist version="1.0">
<dict>
  <key>Label</key>
  <string>${_ACG_CHROME_CDP_LABEL}</string>
  <key>ProgramArguments</key>
  <array>
    <string>/Applications/Google Chrome.app/Contents/MacOS/Google Chrome</string>
    <string>--remote-debugging-port=${_ACG_CHROME_CDP_PORT}</string>
    <string>--user-data-dir=${_ACG_CHROME_CDP_AUTH_DIR}</string>
    <string>--password-store=basic</string>
    <string>--no-first-run</string>
    <string>--no-default-browser-check</string>
  </array>
  <key>RunAtLoad</key>
  <true/>
  <key>KeepAlive</key>
  <true/>
  <key>StandardErrorPath</key>
  <string>/tmp/k3d-manager-chrome-cdp.err</string>
</dict>
</plist>
PLIST
}
```

### Add public function `acg_chrome_cdp_install` (after `acg_watch_stop`)

```bash
function acg_chrome_cdp_install() {
  if [[ "${1:-}" == "-h" || "${1:-}" == "--help" ]]; then
    echo "Usage: acg_chrome_cdp_install"
    echo "Install a launchd agent that starts Chrome with CDP on port ${_ACG_CHROME_CDP_PORT}."
    echo "Chrome uses ${_ACG_CHROME_CDP_AUTH_DIR} as its profile (shared with Playwright scripts)."
    echo "Log: /tmp/k3d-manager-chrome-cdp.err"
    return 0
  fi

  if ! _is_mac; then
    _info "[acg] acg_chrome_cdp_install is macOS only ‚Äî skipping"
    return 0
  fi

  if [[ ! -f "/Applications/Google Chrome.app/Contents/MacOS/Google Chrome" ]]; then
    _err "[acg] Google Chrome not found at /Applications/Google Chrome.app ‚Äî install Chrome first"
  fi

  _acg_chrome_cdp_write_plist

  if launchctl list "${_ACG_CHROME_CDP_LABEL}" >/dev/null 2>&1; then
    _info "[acg] Reloading Chrome CDP agent ${_ACG_CHROME_CDP_LABEL}..."
    launchctl unload "${_ACG_CHROME_CDP_PLIST}" 2>/dev/null || true
  fi

  launchctl load "${_ACG_CHROME_CDP_PLIST}"
  _info "[acg] Chrome CDP agent installed: ${_ACG_CHROME_CDP_LABEL}"
  _info "[acg] Chrome launches on login with --remote-debugging-port=${_ACG_CHROME_CDP_PORT}"
  _info "[acg] Profile: ${_ACG_CHROME_CDP_AUTH_DIR}"
  _info "[acg] Log: /tmp/k3d-manager-chrome-cdp.err"
  _info "[acg] Open http://localhost:${_ACG_CHROME_CDP_PORT} to verify CDP is active"
}
```

### Add public function `acg_chrome_cdp_uninstall` (after `acg_chrome_cdp_install`)

```bash
function acg_chrome_cdp_uninstall() {
  if [[ "${1:-}" == "-h" || "${1:-}" == "--help" ]]; then
    echo "Usage: acg_chrome_cdp_uninstall"
    echo "Unload and remove the Chrome CDP launchd agent."
    return 0
  fi

  if ! _is_mac; then
    _info "[acg] acg_chrome_cdp_uninstall is macOS only ‚Äî skipping"
    return 0
  fi

  if launchctl list "${_ACG_CHROME_CDP_LABEL}" >/dev/null 2>&1; then
    launchctl unload "${_ACG_CHROME_CDP_PLIST}" 2>/dev/null || true
    _info "[acg] Chrome CDP agent stopped: ${_ACG_CHROME_CDP_LABEL}"
  fi

  if [[ -f "${_ACG_CHROME_CDP_PLIST}" ]]; then
    rm -f "${_ACG_CHROME_CDP_PLIST}"
    _info "[acg] Plist removed: ${_ACG_CHROME_CDP_PLIST}"
  fi
}
```
