## Goal

Add `/acg-ask <question>` as a Slack slash command. The user types a natural-language
question about the cluster; the webhook spawns `claude -p` with `--allowedTools Bash`
so Claude can run its own `kubectl` commands to investigate, then posts the answer back
to the Slack thread.

No API key management needed — the `claude` CLI handles auth.

```python
def _run_cluster_ask(job_id, question, response_url, thread_ts=None):
    """Spawn claude -p with the user's question; post the answer to Slack."""
    job_dir = JOB_DIR / job_id
    _ANSI = re.compile(r'\x1b\[[0-9;]*[mK]')

    def _finish(text, status="success"):
        (job_dir / "status").write_text(status)
        (job_dir / "output").write_text(text)
        if thread_ts:
            _post_slack_bot(text, thread_ts=thread_ts)
        elif response_url:
            _slack_post(response_url, text)

    system = (
        "You are troubleshooting a Kubernetes environment. "
        "Available kubectl contexts:\n"
        "  - k3d-k3d-cluster: local k3d cluster (Vault, ArgoCD, Istio, ESO, Keycloak)\n"
        "  - ubuntu-k3s: remote k3s on ACG EC2 (shopping-cart apps, PostgreSQL, Redis, RabbitMQ)\n\n"
        "Use kubectl commands to investigate. Be concise — your answer will be posted to Slack. "
        "Plain text or Slack mrkdwn only. No ANSI codes. No markdown headers."
    )
    prompt = f"{system}\n\nUser question: {question}"

    try:
        result = subprocess.run(
            ["claude", "-p", prompt, "--allowedTools", "Bash"],
            capture_output=True, text=True, timeout=120,
        )
        answer = _ANSI.sub("", result.stdout).strip()
        if not answer:
            answer = _ANSI.sub("", result.stderr).strip() or "No output from Claude."
        _finish(f"🤖 *Claude:* {answer}")
    except subprocess.TimeoutExpired:
        _finish("❌ *Claude* timed out after 120s — try a more specific question", status="failed")
    except FileNotFoundError:
        _finish("❌ `claude` CLI not found — ensure Claude Code is installed and on PATH", status="failed")
    except Exception as exc:
        _finish(f"❌ *Claude error:* {exc}", status="failed")

```
