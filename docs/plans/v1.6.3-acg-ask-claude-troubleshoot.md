## Goal

Add `/ask [claude|gemini|codex] <question>` as a Slack slash command. The first word
selects the AI agent; the rest is the question. Agent defaults to `claude` if omitted.

| Agent | CLI invocation | Best for |
|-------|---------------|----------|
| `claude` | `claude -p "<prompt>" --allowedTools Bash` | Live cluster troubleshooting — runs its own kubectl |
| `gemini` | `gemini --model gemini-2.5-flash --prompt "<prompt>"` | Analysis and diagnosis — same CLI already in webhook |
| `codex` | `codex exec "<prompt>"` | Code questions — how does X work in the repo |

```python
def _run_cluster_ask(job_id, agent, question, response_url, thread_ts=None):
    """Spawn claude/gemini/codex with the user's question; post the answer to Slack."""
    job_dir = JOB_DIR / job_id
    _ANSI = re.compile(r'\x1b\[[0-9;]*[mK]')

    def _finish(text, status="success"):
        (job_dir / "status").write_text(status)
        (job_dir / "output").write_text(text)
        if thread_ts:
            _post_slack_bot(text, thread_ts=thread_ts)
        elif response_url:
            _slack_post(response_url, text)

    system = (
        "You are troubleshooting a Kubernetes environment. "
        "Available kubectl contexts:\n"
        "  - k3d-k3d-cluster: local k3d cluster (Vault, ArgoCD, Istio, ESO, Keycloak)\n"
        "  - ubuntu-k3s: remote k3s on ACG EC2 (shopping-cart apps, PostgreSQL, Redis, RabbitMQ)\n\n"
        "Be concise — your answer will be posted to Slack. "
        "Plain text or Slack mrkdwn only. No ANSI codes. No markdown headers."
    )

    if agent == "claude":
        prompt = f"{system}\nUse kubectl commands to investigate.\n\nUser question: {question}"
        cmd = ["claude", "-p", prompt, "--allowedTools", "Bash"]
    elif agent == "gemini":
        gemini_bin = os.environ.get("K3DM_GEMINI_BIN", "gemini")
        prompt = f"{system}\n\nUser question: {question}"
        cmd = [gemini_bin, "--model", os.environ.get("K3DM_ANALYSIS_MODEL", "gemini-2.5-flash"), "--prompt", prompt]
    else:
        prompt = (
            "You are a code assistant for the k3d-manager repository — a modular Bash utility "
            "for managing local Kubernetes dev clusters (Istio, Vault, Jenkins, OpenLDAP, ESO). "
            "Entry point: scripts/k3d-manager. Plugins in scripts/plugins/. "
            "Be concise — your answer will be posted to Slack.\n\n"
            f"Question: {question}"
        )
        cmd = ["codex", "exec", prompt]

    try:
        result = subprocess.run(cmd, capture_output=True, text=True, timeout=120)
        answer = _ANSI.sub("", result.stdout).strip()
        if not answer:
            answer = _ANSI.sub("", result.stderr).strip() or f"No output from {agent}."
        _finish(f"🤖 *{agent}:* {answer}")
    except subprocess.TimeoutExpired:
        _finish(f"❌ *{agent}* timed out after 120s — try a more specific question", status="failed")
    except FileNotFoundError:
        _finish(f"❌ `{agent}` CLI not found — ensure it is installed and on PATH", status="failed")
    except Exception as exc:
        _finish(f"❌ *{agent} error:* {exc}", status="failed")

```
