Replace hardcoded `_copilot_review` calls in k3d-manager public callers with a new
generic dispatch function `_ai_agent_review` in lib-foundation. The backend is selected
by `AI_REVIEW_FUNC` (no `K3DM_` prefix — this is lib-foundation generic code). Model is
selected by `AI_REVIEW_MODEL`. Both default to the Copilot CLI + `gpt-5.4-mini`.

`_copilot_review` is NOT removed or renamed — it stays as the Copilot-specific
implementation that `_ai_agent_review` routes to.

Add this function immediately after the closing `}` of `_copilot_review` (currently ends around line 1720):

```bash
function _ai_agent_review() {
   local ai_func="${AI_REVIEW_FUNC:-copilot}"
   local model="${AI_REVIEW_MODEL:-gpt-5.4-mini}"

   case "$ai_func" in
      copilot) _copilot_review --model "$model" "$@" ;;
      *) _err "Unknown AI_REVIEW_FUNC: ${ai_func}. Supported: copilot" ;;
   esac
}
```

#### File 3: `scripts/tests/lib/ai_agent_review.bats` — NEW BATS file

Create `scripts/tests/lib/ai_agent_review.bats` with these tests:

```bash
#!/usr/bin/env bats

bats_require_minimum_version 1.5.0

setup() {
  source "${BATS_TEST_DIRNAME}/../test_helpers.bash"
  init_test_env
  source "${SCRIPT_DIR}/lib/system.sh"
}

@test "_ai_agent_review dispatches to _copilot_review with default model" {
  export_stubs
  export K3DM_ENABLE_AI=1
  unset AI_REVIEW_FUNC AI_REVIEW_MODEL
  _copilot_review() { printf '%s\n' "$*" >> "$RUN_LOG"; }
  export -f _copilot_review

  run _ai_agent_review --prompt "hello"
  [ "$status" -eq 0 ]
  grep -q -- "--model gpt-5.4-mini" "$RUN_LOG"
  grep -q -- "--prompt" "$RUN_LOG"
}

@test "_ai_agent_review uses AI_REVIEW_MODEL when set" {
  export_stubs
  export K3DM_ENABLE_AI=1
  export AI_REVIEW_FUNC=copilot
  export AI_REVIEW_MODEL=claude-sonnet-4-6
  _copilot_review() { printf '%s\n' "$*" >> "$RUN_LOG"; }
  export -f _copilot_review

  run _ai_agent_review --prompt "hello"
  [ "$status" -eq 0 ]
  grep -q -- "--model claude-sonnet-4-6" "$RUN_LOG"
}

@test "_ai_agent_review errors on unknown AI_REVIEW_FUNC" {
  export AI_REVIEW_FUNC=unknown_backend
  run _ai_agent_review --prompt "hello"
  [ "$status" -ne 0 ]
  [[ "$output" == *"Unknown AI_REVIEW_FUNC"* ]]
}
```
