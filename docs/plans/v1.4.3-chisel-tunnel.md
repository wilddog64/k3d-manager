## Problem

`_setup_vault_bridge` SSHes into the remote EC2/GCP node to install socat and write a
systemd unit. `autossh` then maintains an SSH-based tunnel for both the k3s API and
the Vault reverse bridge.

Some organisations block outbound SSH (port 22). The SSH dependency exists at two layers:

| Layer | Current | Problem |
|-------|---------|---------|
| Setup | `ssh -i key host bash << REMOTE` | Requires outbound port 22 |
| Runtime k3s API | `autossh -L 6443:localhost:6443` | SSH-based forward tunnel |
| Runtime Vault bridge | `autossh -R 8200:127.0.0.1:18200` + socat | SSH-based reverse tunnel + socat rebind |

**[chisel](https://github.com/jpillora/chisel)** (Go, single binary, Apache-2.0) replaces all
three layers with HTTPS WebSocket connections on a configurable port (default 8080).
The remote node runs `chisel server` (started via cloud-init / SSM — no SSH).
The Mac runs `chisel client`, making a single outbound HTTPS connection that carries both
the k3s API forward tunnel and the Vault reverse tunnel simultaneously. No socat needed
because chisel binds to `0.0.0.0` natively.

---

## Architecture

```
Mac (chisel client)           EC2/GCP node (chisel server :8080)
    │                                  │
    │── HTTPS WS ──────────────────────┤
    │                                  │
    │  localhost:6443 ──forward──▶  :6443 (k3s API)
    │  localhost:18200 ◀──reverse──  0.0.0.0:8201 (Vault, reachable by pods)
```

- **k3s API**: local forward — Mac opens `localhost:6443`, proxied to EC2 `localhost:6443`
- **Vault bridge**: remote forward — EC2 server opens `0.0.0.0:8201`, proxied back to Mac
  `localhost:18200` (where `kubectl port-forward` exposes Hub Vault)
- No socat, no autossh, no SSH at runtime

## Dependency Management

**Pinned version:** `1.10.1` — stored in `_CHISEL_VERSION` default, overridable via env.

**Checksum verification:** Before running any chisel binary, verify SHA256 against
values pinned in `_chisel_expected_sha256()`. Codex must fetch these from:
```
https://github.com/jpillora/chisel/releases/download/v1.10.1/checksums.txt
```
and fill them into the lookup table before committing. Do not commit placeholder values.

**Install strategy (local Mac):** `brew install chisel` first; fall back to GitHub releases
binary download + checksum verify + gunzip.

**Install strategy (remote node):**
- AWS: `ssm_exec` (HTTPS to SSM endpoint — no SSH) at deploy time
- GCP: `startup-script` instance metadata — chisel server installed at first boot by cloud-init

**Version upgrade path:** Change `_CHISEL_VERSION` default + update SHA256 lookup table.
Nothing else to change.

## Change 1 — Create `scripts/plugins/chisel.sh`

```bash
#!/usr/bin/env bash
# scripts/plugins/chisel.sh — chisel HTTPS tunnel: dependency management + lifecycle

set -euo pipefail

_CHISEL_VERSION="${CHISEL_VERSION:-1.10.1}"
_CHISEL_PORT="${CHISEL_PORT:-8080}"
_CHISEL_PID_FILE="${HOME}/.local/share/k3d-manager/chisel-client.pid"
_CHISEL_LOG_FILE="${HOME}/.local/share/k3d-manager/chisel-client.log"

function _chisel_expected_sha256() {
  local os="$1" arch="$2"
  # Fill these from checksums.txt at the pinned version release page:
  # https://github.com/jpillora/chisel/releases/download/v${_CHISEL_VERSION}/checksums.txt
  # The .gz file checksums (not the raw binary) are listed there.
  case "${os}_${arch}" in
    linux_amd64)  printf '%s' "FILL_FROM_CHECKSUMS_TXT" ;;
    linux_arm64)  printf '%s' "FILL_FROM_CHECKSUMS_TXT" ;;
    darwin_amd64) printf '%s' "FILL_FROM_CHECKSUMS_TXT" ;;
    darwin_arm64) printf '%s' "FILL_FROM_CHECKSUMS_TXT" ;;
    *)            printf '' ;;
  esac
}

function _ensure_chisel() {
  if _command_exist chisel; then
    _info "[chisel] chisel already installed: $(chisel --version 2>&1 | head -n1)"
    return 0
  fi

  if _command_exist brew; then
    _info "[chisel] Installing chisel via brew..."
    _run_command -- brew install chisel
    if _command_exist chisel; then return 0; fi
  fi

  local os arch tmp_gz
  os="$(uname -s | tr '[:upper:]' '[:lower:]')"
  arch="$(uname -m)"
  case "${arch}" in
    x86_64)           arch="amd64" ;;
    aarch64|arm64)    arch="arm64" ;;
  esac

  local url="https://github.com/jpillora/chisel/releases/download/v${_CHISEL_VERSION}/chisel_${_CHISEL_VERSION}_${os}_${arch}.gz"
  tmp_gz=$(mktemp)
  trap 'rm -f "${tmp_gz}"' EXIT

  _info "[chisel] Downloading chisel ${_CHISEL_VERSION} (${os}/${arch})..."
  if ! curl -fsSL "${url}" -o "${tmp_gz}"; then
    _err "[chisel] Failed to download chisel from ${url}"
    return 1
  fi

  local expected
  expected="$(_chisel_expected_sha256 "${os}" "${arch}")"
  if [[ -n "${expected}" ]]; then
    _info "[chisel] Verifying SHA256..."
    if ! echo "${expected}  ${tmp_gz}" | sha256sum --check --status 2>/dev/null; then
      _err "[chisel] SHA256 mismatch for chisel ${_CHISEL_VERSION} ${os}/${arch} — aborting"
      return 1
    fi
  else
    _info "[chisel] WARNING: no pinned checksum for ${os}/${arch} — skipping verification"
  fi

  local install_dir="/usr/local/bin"
  local tmp_bin
  tmp_bin=$(mktemp)
  gunzip -c "${tmp_gz}" > "${tmp_bin}"
  chmod +x "${tmp_bin}"
  _run_command --prefer-sudo -- mv "${tmp_bin}" "${install_dir}/chisel"
  _info "[chisel] chisel ${_CHISEL_VERSION} installed to ${install_dir}/chisel"
}

function _chisel_remote_install_ssm() {
  local instance_id="$1"
  local chisel_port="${_CHISEL_PORT}"
  local chisel_version="${_CHISEL_VERSION}"

  _info "[chisel] Installing chisel server ${chisel_version} on ${instance_id} via SSM..."

  local cmd
  cmd=$(cat <<EOF
set -euo pipefail
CHISEL_VERSION="${chisel_version}"
CHISEL_PORT="${chisel_port}"
if command -v chisel >/dev/null 2>&1 && chisel --version 2>&1 | grep -q "\${CHISEL_VERSION}"; then
  echo "chisel already at \${CHISEL_VERSION} — skipping download"
else
  curl -fsSL "https://github.com/jpillora/chisel/releases/download/v\${CHISEL_VERSION}/chisel_\${CHISEL_VERSION}_linux_amd64.gz" \
    -o /tmp/chisel.gz
  gunzip -c /tmp/chisel.gz > /usr/local/bin/chisel
  chmod +x /usr/local/bin/chisel
  rm -f /tmp/chisel.gz
fi
tee /etc/systemd/system/chisel-server.service > /dev/null << 'UNIT'
[Unit]
Description=chisel HTTPS tunnel server
After=network.target

[Service]
ExecStart=/usr/local/bin/chisel server --port ${chisel_port} --reverse
Restart=always
RestartSec=5

[Install]
WantedBy=multi-user.target
UNIT
systemctl daemon-reload
systemctl enable chisel-server
systemctl restart chisel-server
echo "chisel-server started on port ${chisel_port}"
EOF
)

  ssm_exec "${instance_id}" "${cmd}"
  _info "[chisel] chisel server active on ${instance_id}:${chisel_port}"
}

function chisel_tunnel_start() {
  if [[ "${1:-}" == "-h" || "${1:-}" == "--help" ]]; then
    cat <<'HELP'
Usage: chisel_tunnel_start

Start chisel client on the Mac side. Creates two tunnels over a single HTTPS connection:
  localhost:6443    → server:6443     (k3s API forward)
  0.0.0.0:8201      ← localhost:18200 (Vault reverse bridge)

Config (env overrides):
  TUNNEL_SSH_HOST         Chisel server hostname (same var as autossh — default: ubuntu)
  CHISEL_PORT             Chisel server port (default: 8080)
  TUNNEL_LOCAL_PORT       k3s API local port (default: 6443)
  TUNNEL_REMOTE_PORT      k3s API remote port (default: 6443)
  TUNNEL_VAULT_REMOTE_PORT Vault bridge remote port on EC2 (default: 8201)
  TUNNEL_VAULT_LOCAL_PORT  Vault local port on Mac (default: 18200)
HELP
    return 0
  fi

  if [[ -f "${_CHISEL_PID_FILE}" ]]; then
    local existing_pid
    existing_pid=$(cat "${_CHISEL_PID_FILE}")
    if kill -0 "${existing_pid}" 2>/dev/null; then
      _info "[chisel] Tunnel already running (PID ${existing_pid}) — skipping"
      return 0
    fi
    rm -f "${_CHISEL_PID_FILE}"
  fi

  _ensure_chisel || return 1

  local server_host="${TUNNEL_SSH_HOST:-ubuntu}"
  local chisel_port="${_CHISEL_PORT}"
  local local_port="${TUNNEL_LOCAL_PORT:-6443}"
  local remote_port="${TUNNEL_REMOTE_PORT:-6443}"
  local vault_remote_port="${TUNNEL_VAULT_REMOTE_PORT:-8201}"
  local vault_local_port="${TUNNEL_VAULT_LOCAL_PORT:-18200}"

  mkdir -p "$(dirname "${_CHISEL_PID_FILE}")" "$(dirname "${_CHISEL_LOG_FILE}")"

  _info "[chisel] Starting tunnel → ${server_host}:${chisel_port}"
  _info "[chisel]   forward: localhost:${local_port} → server:${remote_port} (k3s API)"
  _info "[chisel]   reverse: 0.0.0.0:${vault_remote_port} → localhost:${vault_local_port} (Vault)"

  chisel client \
    --keepalive 30s \
    "https://${server_host}:${chisel_port}" \
    "localhost:${local_port}:localhost:${remote_port}" \
    "R:0.0.0.0:${vault_remote_port}:localhost:${vault_local_port}" \
    >> "${_CHISEL_LOG_FILE}" 2>&1 &

  local pid=$!
  printf '%s\n' "${pid}" > "${_CHISEL_PID_FILE}"
  _info "[chisel] Tunnel started (PID ${pid}). Log: ${_CHISEL_LOG_FILE}"

  sleep 2
  if ! kill -0 "${pid}" 2>/dev/null; then
    _err "[chisel] chisel client exited immediately — check ${_CHISEL_LOG_FILE}"
    rm -f "${_CHISEL_PID_FILE}"
    return 1
  fi
}

function chisel_tunnel_stop() {
  if [[ ! -f "${_CHISEL_PID_FILE}" ]]; then
    _info "[chisel] No PID file found — tunnel not running"
    return 0
  fi
  local pid
  pid=$(cat "${_CHISEL_PID_FILE}")
  if kill -0 "${pid}" 2>/dev/null; then
    _info "[chisel] Stopping chisel client (PID ${pid})..."
    kill "${pid}" 2>/dev/null || true
  fi
  rm -f "${_CHISEL_PID_FILE}"
  _info "[chisel] Tunnel stopped."
}
```

```bash
    _info "[k3s_remote] TUNNEL_PROVIDER=chisel — skipping socat vault-bridge (chisel handles it)"
  fi

  _info "[k3s_remote] k3s install complete."
```
