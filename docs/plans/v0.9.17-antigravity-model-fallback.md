### Step 1 — Add model list constant and helper (before `_ensure_antigravity`)

Add at line 16 (before `function _ensure_antigravity`):

```bash
_ANTIGRAVITY_GEMINI_MODELS=(
  "gemini-1.5-flash"
  "gemini-2.0-flash"
  "gemini-2.5-flash"
)

function _antigravity_gemini_prompt() {
  local prompt="$1"
  local output exit_code

  for model in "${_ANTIGRAVITY_GEMINI_MODELS[@]}"; do
    _info "Trying gemini model: ${model}..."
    sleep 2
    output=$(gemini --model "$model" --prompt "$prompt" 2>&1)
    exit_code=$?
    if [[ $exit_code -eq 0 ]]; then
      echo "$output"
      return 0
    fi
    if [[ "$output" == *"429"* || "$output" == *"RESOURCE_EXHAUSTED"* || "$output" == *"rateLimitExceeded"* ]]; then
      _info "Model ${model} capacity exhausted — trying next model..."
      continue
    fi
    echo "$output"
    return "$exit_code"
  done

  _err "All gemini models exhausted (429). Models tried: ${_ANTIGRAVITY_GEMINI_MODELS[*]}"
}
```

### Add these three tests

```bash
@test "_antigravity_gemini_prompt: succeeds on first model" {
  _info() { :; }
  gemini() {
    [[ "$2" == "gemini-1.5-flash" ]] && echo "ok" && return 0
    return 1
  }
  export -f _info gemini
  source scripts/plugins/antigravity.sh

  run _antigravity_gemini_prompt "test prompt"
  [ "$status" -eq 0 ]
  [[ "$output" == *"ok"* ]]
  unset -f _info gemini
}

@test "_antigravity_gemini_prompt: falls back to next model on 429" {
  _info() { :; }
  call=0
  gemini() {
    call=$((call + 1))
    if [[ $call -eq 1 ]]; then
      echo "429 RESOURCE_EXHAUSTED rateLimitExceeded"
      return 1
    fi
    echo "ok"
    return 0
  }
  export -f _info gemini
  source scripts/plugins/antigravity.sh

  run _antigravity_gemini_prompt "test prompt"
  [ "$status" -eq 0 ]
  [[ "$output" == *"ok"* ]]
  unset -f _info gemini
}

@test "_antigravity_gemini_prompt: fails when all models exhausted" {
  _info() { :; }
  _err() { echo "$*"; exit 1; }
  gemini() { echo "429 RESOURCE_EXHAUSTED rateLimitExceeded"; return 1; }
  export -f _info _err gemini
  source scripts/plugins/antigravity.sh

  run _antigravity_gemini_prompt "test prompt"
  [ "$status" -ne 0 ]
  [[ "$output" == *"All gemini models exhausted"* ]]
  unset -f _info _err gemini
}
```
