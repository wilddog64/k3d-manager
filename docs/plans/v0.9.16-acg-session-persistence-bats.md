### Setup

```bash
#!/usr/bin/env bats
# shellcheck shell=bash disable=SC1091,SC2329

setup() {
  ANTIGRAVITY_LIB="${BATS_TEST_DIRNAME}/../../plugins/antigravity.sh"
  # shellcheck source=/dev/null
  source "$ANTIGRAVITY_LIB"
}

bats_require_minimum_version 1.5.0
```

### Test 1 — `_acg_load_session`: missing file returns 1

```bash
@test "_acg_load_session: missing session file returns 1" {
  _warn() { :; }
  export -f _warn
  _ACG_SESSION_FILE="/tmp/nonexistent_acg_session_$$.json"

  run _acg_load_session
  [ "$status" -eq 1 ]
  unset -f _warn
}
```

### Test 2 — `_acg_load_session`: existing file prints path

```bash
@test "_acg_load_session: existing session file prints path" {
  session_file="$(mktemp -t acg_session.XXXXXX)"
  printf '{"cookies":[],"origins":[]}' > "$session_file"
  _ACG_SESSION_FILE="$session_file"

  run _acg_load_session
  [ "$status" -eq 0 ]
  [ "$output" = "$session_file" ]
  rm -f "$session_file"
}
```

### Test 3 — `_acg_load_session`: empty file returns 1

```bash
@test "_acg_load_session: empty session file returns 1" {
```

### Test 4 — `acg_login`: calls `_antigravity_launch` and `_acg_save_session`

```bash
@test "acg_login: calls _antigravity_launch and _acg_save_session" {
  launch_called=0
  save_called=0
  _ensure_antigravity() { return 0; }
  _antigravity_launch() { launch_called=1; return 0; }
  _acg_save_session() { save_called=1; return 0; }
  export -f _ensure_antigravity _antigravity_launch _acg_save_session

  run acg_login
  [ "$status" -eq 0 ]
  unset -f _ensure_antigravity _antigravity_launch _acg_save_session
}
```

### Test 5 — `antigravity_acg_extend`: exits 1 when session missing

```bash
@test "antigravity_acg_extend: exits 1 when session file missing" {
  _ensure_antigravity() { return 0; }
```
