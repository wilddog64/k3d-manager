# Feature: v1.6.0 вЂ” Slack thread commands

**Branch:** `k3d-manager-v1.6.0`
**Files:** `bin/k3dm-webhook`, `Makefile`

---

## Problem

Webhook job notifications are now threaded in `#acg-automation`. Users can watch
progress but cannot interact вЂ” to kill a hung job or request a re-diagnosis they must
make a raw HTTP call. Adding thread-reply commands keeps all interaction scoped to the
job's thread.

---

## Solution

Subscribe the k3d-manager Slack bot to `message.channels` events via the Slack Events
API. When a reply arrives in a known job thread, parse the first word as a command and
dispatch it. All replies stay in that job's thread via `_notify_job`.

**Supported commands (case-insensitive, leading `/` optional):**

| Command | Action |
|---------|--------|
| `kill` | Kill the running job process; trigger cleanup if acg-up |
| `diagnosis` | Re-run `_analyze_failure` on last job log; post result to thread |
| `status` | Post current job state + elapsed time to thread |
| `logs` | Post last 20 non-empty log lines to thread |

---

## Slack App changes (manual вЂ” one-time)

1. In the Slack app settings в†’ **Event Subscriptions** в†’ enable.
2. Set Request URL to `https://webhook.3ai-talk.org/slack/events`.
3. Subscribe to bot event: `message.channels`.
4. Under **OAuth & Permissions** add bot scope: `channels:history`.
5. Reinstall the app to the workspace.
6. Copy the **Signing Secret** from **Basic Information**.

```python
def _verify_slack_signature(raw_body, timestamp, signature):
    """Return True if the Slack request signature is valid."""
    if not SLACK_SIGNING_SECRET:
        return False
    import time
    if abs(time.time() - int(timestamp)) > 300:
        return False
    base = f"v0:{timestamp}:{raw_body.decode()}"
    expected = "v0=" + hmac.new(
        SLACK_SIGNING_SECRET.encode(),
        base.encode(),
        digestmod="sha256",
    ).hexdigest()
    return hmac.compare_digest(expected, signature)


def _find_job_by_thread_ts(thread_ts):
    """Return job_id whose thread_ts file matches, or None."""
    if not JOB_DIR.exists():
        return None
    for ts_file in JOB_DIR.glob("*/thread_ts"):
        try:
            if ts_file.read_text().strip() == thread_ts:
                return ts_file.parent.name
        except OSError:
            pass
    return None


def _handle_thread_command(job_id, command):
    """Dispatch a thread command for job_id. Runs in a background thread."""
    cmd = command.strip().lstrip("/").lower().split()[0] if command.strip() else ""

    if cmd == "kill":
        proc = _running_procs.get(job_id)
        if proc:
            try:
                import os as _os
                _os.killpg(_os.getpgid(proc.pid), __import__("signal").SIGTERM)
            except Exception:
                pass
            _notify_job(job_id, f"рџ›‘ *Kill requested* вЂ” job `{job_id}` signal sent")
        else:
            _notify_job(job_id, f"вљ пёЏ Job `{job_id}` is not running вЂ” nothing to kill")

    elif cmd == "diagnosis":
        log_file = JOB_DIR / job_id / "log"
        if log_file.exists():
            lines = log_file.read_text().splitlines(keepends=True)
            try:
                result = _analyze_failure(lines)
                _notify_job(job_id, f"рџ”Ќ *Re-diagnosis* (job `{job_id}`)\n{result}")
            except Exception as exc:
                _notify_job(job_id, f"вљ пёЏ Diagnosis failed: {exc}")
        else:
            _notify_job(job_id, f"вљ пёЏ No log found for job `{job_id}`")

    elif cmd == "status":
        status_file = JOB_DIR / job_id / "status"
        status = status_file.read_text().strip() if status_file.exists() else "unknown"
        _notify_job(job_id, f"в„№пёЏ *Job `{job_id}`* status: `{status}`")

    elif cmd == "logs":
        log_file = JOB_DIR / job_id / "log"
        if log_file.exists():
            last = [l.rstrip() for l in log_file.read_text().splitlines() if l.strip()][-20:]
            _notify_job(job_id, f"рџ“‹ *Last 20 lines* (job `{job_id}`)\n```{''.join(last)}```")
        else:
            _notify_job(job_id, f"вљ пёЏ No log found for job `{job_id}`")

    else:
        _notify_job(
            job_id,
            f"вќ“ Unknown command `{command.strip()}` вЂ” available: `kill`, `diagnosis`, `status`, `logs`",
```

## Change 4 вЂ” `bin/k3dm-webhook`: add `/slack/events` route in `do_POST`

The `/slack/events` endpoint must be **before** `_auth()` вЂ” Slack's URL verification
challenge has no Bearer token. Insert before the `if not self._auth()` check.

**Exact old block:**


**Exact new block:**
```python
    def do_POST(self):
        if self.path == "/slack/events":
            length = int(self.headers.get("Content-Length", 0))
            raw_body = self.rfile.read(min(length, MAX_BODY))
            timestamp = self.headers.get("X-Slack-Request-Timestamp", "0")
            signature = self.headers.get("X-Slack-Signature", "")
            if not _verify_slack_signature(raw_body, timestamp, signature):
                self._json(401, {"error": "invalid signature"})
                return
            try:
                event = json.loads(raw_body)
            except Exception:
                self._json(400, {"error": "bad json"})
                return
            if event.get("type") == "url_verification":
                self._json(200, {"challenge": event.get("challenge", "")})
                return
            ev = event.get("event", {})
            if (
                ev.get("type") == "message"
                and not ev.get("bot_id")
                and not ev.get("subtype")
                and ev.get("thread_ts")
                and ev.get("thread_ts") != ev.get("ts")
            ):
                thread_ts = ev["thread_ts"]
                text = ev.get("text", "").strip()
                job_id = _find_job_by_thread_ts(thread_ts)
                if job_id and text:
                    threading.Thread(
                        target=_handle_thread_command,
                        args=(job_id, text),
                        daemon=True,
                    ).start()
            self._json(200, {"ok": True})
            return

        if not self._auth():
            self._json(401, {"error": "unauthorized"})
            return
        if self.path.startswith("/api/") and not self.path.startswith("/api/v1/"):
```
