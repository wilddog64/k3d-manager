## Problem

When `acg-up` (triggered from Slack via webhook) stalls mid-run — e.g. a `kubectl wait`
hangs, a CloudFormation step blocks, or any other indefinite wait — the webhook thread
stays alive indefinitely and the user has no signal other than the 2-minute progress
tick showing the same last 3 lines of output.

Desired behaviour: after 6 minutes of no new output, the webhook (1) calls Claude Haiku
to classify whether the process is genuinely stuck, (2) posts the verdict to Slack with
a kill instruction, and (3) exposes a `kill` action on `/api/v1/cluster` so the user
can terminate the stuck process from Slack.

```python
def _analyze_stall(lines):
    """Feed last 20 non-empty lines to Claude Haiku; return verdict string."""
    last_20 = [l.rstrip() for l in lines[-20:] if l.strip()]
    last_output = "\n".join(last_20)
    prompt = (
        "acg-up has been running and produced no new output for several minutes. "
        f"Last 20 lines of output:\n\n{last_output}\n\n"
        "Is this process genuinely stalled (no progress, waiting on something that "
        "will not resolve), or is it still doing legitimate work (e.g. waiting for a "
        "pod to become Ready)? "
        "Reply with exactly: STALLED or PROGRESSING, then one sentence why. Be brief."
    )
    return _call_claude(prompt)
```

```python
def _run_cluster(job_id, action, provider="aws"):
    job_dir = JOB_DIR / job_id
    lines = []
    _provider_map = {"aws": "k3s-aws", "gcp": "k3s-gcp", "azure": "k3s-azure"}
    _cluster_provider = _provider_map.get(provider, "k3s-aws")
    _start = datetime.datetime.utcnow().timestamp()
    _progress_timer = None
    _last_line_count = 0
    _stall_tick_count = 0
    _stall_analyzed = False

    def _log(msg):
        lines.append(msg + "\n")

    def _notify(text):
        url = SLACK_WEBHOOK_URL
        if not url:
            _ru_file = job_dir / "response_url"
            if _ru_file.exists():
                url = _ru_file.read_text().strip()
        if url:
            _slack_post(url, text)

    def _post_progress():
        nonlocal _progress_timer, _last_line_count, _stall_tick_count, _stall_analyzed
        elapsed = int((datetime.datetime.utcnow().timestamp() - _start) / 60)
        tail = [l.rstrip() for l in lines[-3:] if l.strip()]
        tail_text = "\n".join(tail) if tail else "—"
        _notify(f"⏳ *acg-{action}* ({provider}) still running — {elapsed}m elapsed\n```{tail_text}```")
        current_count = len(lines)
        if current_count == _last_line_count:
            _stall_tick_count += 1
        else:
            _stall_tick_count = 0
            _stall_analyzed = False
        _last_line_count = current_count
        if _stall_tick_count >= _STALL_TICKS_THRESHOLD and not _stall_analyzed:
            _stall_analyzed = True
            analysis = _analyze_stall(lines)
            _notify(
                f"⚠️ *Stall detected* — no new output for {_stall_tick_count * 2}m "
                f"(job `{job_id}`)\n*AI:* {analysis}\n"
                f'To kill: POST `/api/v1/cluster` `{{"action":"kill","job_id":"{job_id}"}}`'
            )
        _progress_timer = threading.Timer(120, _post_progress)
        _progress_timer.daemon = True
        _progress_timer.start()

    def _finish(status):
        if _progress_timer:
            _progress_timer.cancel()
        with _running_procs_lock:
            _running_procs.pop(job_id, None)
        (job_dir / "status").write_text(status)
        (job_dir / "output").write_text("".join(lines))
        emoji = "✅" if status == "success" else "❌"
        elapsed = int((datetime.datetime.utcnow().timestamp() - _start) / 60)
        summary = f"{emoji} *acg-{action}* ({provider}) — {status} in {elapsed}m"
        _notify(summary)

    try:
        if action == "up":
            cmd = ["make", "up", f"CLUSTER_PROVIDER={_cluster_provider}"]
        else:
            cmd = ["make", "down", "KEEP_LOCAL=1"]
        _log(f"Running {' '.join(cmd)}...")
        _progress_timer = threading.Timer(120, _post_progress)
        _progress_timer.daemon = True
        _progress_timer.start()
        proc = subprocess.Popen(
            cmd,
            cwd=REPO_ROOT,
            stdout=subprocess.PIPE,
            stderr=subprocess.STDOUT,
            text=True
        )
        with _running_procs_lock:
            _running_procs[job_id] = proc
        for line in proc.stdout:
            lines.append(line)
        proc.wait()
        if proc.returncode != 0:
            raise RuntimeError(f"{' '.join(cmd)} exited {proc.returncode}")
        _log(f"{action} complete")
        if action == "up":
            _record_acg_state(provider)
        _finish("success")
    except Exception as exc:
        _log(f"ERROR: {exc}")
        _finish("failed")
```

**Exact new block:**

```python
        if self.path == "/api/v1/cluster":
            action = body.get("action", "")
            if action == "kill":
                job_id = body.get("job_id", "")
                if not job_id or not _JOB_ID_RE.match(job_id):
                    self._json(400, {"error": "job_id required"})
                    return
                with _running_procs_lock:
                    proc = _running_procs.get(job_id)
                if proc:
                    proc.kill()
                    job_dir = JOB_DIR / job_id
                    (job_dir / "status").write_text("killed")
                    self._json(200, {"status": "killed", "job_id": job_id})
                else:
                    self._json(404, {"error": "no running job", "job_id": job_id})
                return
            if action not in ("up", "down"):
                self._json(400, {"error": "action (up|down|kill) required"})
```
