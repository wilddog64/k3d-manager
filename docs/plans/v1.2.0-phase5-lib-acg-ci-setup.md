## File 1: `.github/workflows/ci.yml`

Create directories `.github/workflows/` and write:

```yaml
---
name: CI

"on":
  pull_request:
    types: [opened, synchronize, reopened]
    branches:
      - main

permissions:
  contents: read

concurrency:
  group: ci-${{ github.workflow }}-${{ github.ref }}
  cancel-in-progress: true

jobs:
  lint:
    if: >-
      ${{ github.event.pull_request.head.repo.full_name == github.repository }}
    runs-on: ubuntu-latest
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Setup dependencies
        uses: ./.github/actions/setup

      - name: Run shellcheck
        shell: bash
        run: |
          set -euo pipefail
          mapfile -t files < <(
            find scripts -type f -name '*.sh' \
              -not -path 'scripts/lib/foundation/*' \
              -exec awk 'FNR==1 && /^#!.*bash/ {print FILENAME}' {} +
          )
          if [[ ${#files[@]} -gt 0 ]]; then
            shellcheck -S error "${files[@]}"
          fi

      - name: Run node syntax checks
        shell: bash
        run: |
          set -euo pipefail
          find playwright -type f -name '*.js' -print0 |
          while IFS= read -r -d '' file; do
            node --check "$file"
          done

      - name: Run yamllint on workflow files
        shell: bash
        run: |
          set -euo pipefail
          yamllint .github/workflows/*.yml
```

## File 2: `.github/actions/setup/action.yml`

Create directory `.github/actions/setup/` and write:

```yaml
name: Setup CI Dependencies
description: Install shellcheck, Node.js, and yamllint
runs:
  using: composite
  steps:
    - name: Install apt packages
      shell: bash
      run: |
        set -euo pipefail
        sudo apt-get update
        sudo apt-get install -y shellcheck python3-pip

    - name: Setup Node.js
      uses: actions/setup-node@v4
      with:
        node-version: '20'

    - name: Install yamllint
      shell: bash
      run: |
        set -euo pipefail
        python3 -m pip install --user yamllint
        echo "$HOME/.local/bin" >> "$GITHUB_PATH"
```

## File 3: `scripts/hooks/pre-commit`

```bash
#!/usr/bin/env bash
set -euo pipefail

# Guard: block direct edits to the lib-foundation subtree
# Bypass: set K3DM_SUBTREE_SYNC=1 when running git subtree pull/add
if [[ "${K3DM_SUBTREE_SYNC:-0}" != "1" ]]; then
  subtree_changes="$(git diff --cached --name-only | grep '^scripts/lib/foundation/' || true)"
  if [[ -n "$subtree_changes" ]]; then
    echo "Pre-commit: direct edits to scripts/lib/foundation/ are not allowed." >&2
    echo "Edit upstream in lib-foundation, then run:" >&2
    echo "  K3DM_SUBTREE_SYNC=1 git subtree pull --prefix=scripts/lib/foundation ..." >&2
    echo "" >&2
    echo "Blocked files:" >&2
    echo "$subtree_changes" | sed 's/^/  /' >&2
    exit 1
  fi
fi

# shellcheck on staged .sh files (skip foundation subtree)
staged_sh="$(git diff --cached --name-only --diff-filter=ACM | \
  grep '\.sh$' | grep -v '^scripts/lib/foundation/' || true)"
if [[ -n "$staged_sh" ]]; then
  echo "$staged_sh" | xargs shellcheck -S warning
fi

# node --check on staged .js files
staged_js="$(git diff --cached --name-only --diff-filter=ACM | grep '\.js$' || true)"
if [[ -n "$staged_js" ]]; then
  echo "$staged_js" | xargs node --check
fi
```

After writing the file, make it executable:
```bash
chmod +x /Users/cliang/src/gitrepo/personal/lib-acg/scripts/hooks/pre-commit
```

Also install the hook for the local checkout:
```bash
ln -sf ../../scripts/hooks/pre-commit \
  /Users/cliang/src/gitrepo/personal/lib-acg/.git/hooks/pre-commit
```
