## Problem

`localhost:19090` (Prometheus port-forward to `ubuntu-k3s`) has no LaunchAgent.
Every other persistent port-forward (`argocd`, `frontend`, `vault`, `keycloak`) has a
`KeepAlive=true` plist that auto-restarts when the process dies. Prometheus has none, so
the port-forward dies after each SSH reconnect or OS sleep, and the smoke test reports
`Connection refused` on every subsequent post-provision check.

**Root cause:** The Prometheus port-forward was started manually and never codified as a
LaunchAgent template with a Makefile install target.

---

## Fix

Add a plist template and Makefile install/uninstall targets following the same pattern as
`vault-port-forward`.

### Change 1 — new file `scripts/etc/launchd/com.k3d-manager.prometheus-port-forward.plist.tmpl`

Create this file with exactly this content:

```xml
<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE plist PUBLIC "-//Apple//DTD PLIST 1.0//EN"
  "http://www.apple.com/DTDs/PropertyList-1.0.dtd">
<plist version="1.0">
<dict>
  <key>Label</key>
  <string>com.k3d-manager.prometheus-port-forward</string>
  <key>ProgramArguments</key>
  <array>
    <string>{{KUBECTL_PATH}}</string>
    <string>port-forward</string>
    <string>svc/prometheus-operated</string>
    <string>19090:9090</string>
    <string>-n</string>
    <string>monitoring</string>
    <string>--context</string>
    <string>ubuntu-k3s</string>
  </array>
  <key>EnvironmentVariables</key>
  <dict>
    <key>KUBECONFIG</key>
    <string>{{HOME}}/.kube/config:{{HOME}}/.kube/ubuntu-k3s.yaml:{{HOME}}/.kube/k3s-ubuntu.yaml</string>
  </dict>
  <key>RunAtLoad</key>
  <true/>
  <key>KeepAlive</key>
  <true/>
  <key>StandardOutPath</key>
  <string>{{HOME}}/Library/Logs/k3dm-prometheus-port-forward.log</string>
  <key>StandardErrorPath</key>
  <string>{{HOME}}/Library/Logs/k3dm-prometheus-port-forward.log</string>
</dict>
</plist>
```

**Exact new block (append immediately after the block above):**

```makefile
uninstall-vault-port-forward:
	launchctl bootout "gui/$$(id -u)/com.k3d-manager.vault-port-forward" 2>/dev/null || true
	rm -f "$(HOME)/Library/LaunchAgents/com.k3d-manager.vault-port-forward.plist"
	@echo "Vault port-forward agent removed"

install-prometheus-port-forward:
	sed \
	  -e "s|{{KUBECTL_PATH}}|$$(command -v kubectl)|g" \
	  -e "s|{{HOME}}|$(HOME)|g" \
	  scripts/etc/launchd/com.k3d-manager.prometheus-port-forward.plist.tmpl \
	  > "$(HOME)/Library/LaunchAgents/com.k3d-manager.prometheus-port-forward.plist"
	launchctl bootout "gui/$$(id -u)/com.k3d-manager.prometheus-port-forward" 2>/dev/null || true
	launchctl bootstrap "gui/$$(id -u)" \
	  "$(HOME)/Library/LaunchAgents/com.k3d-manager.prometheus-port-forward.plist"
	@echo "Prometheus port-forward agent installed — port 19090 will stay open while ubuntu-k3s is reachable"

uninstall-prometheus-port-forward:
	launchctl bootout "gui/$$(id -u)/com.k3d-manager.prometheus-port-forward" 2>/dev/null || true
	rm -f "$(HOME)/Library/LaunchAgents/com.k3d-manager.prometheus-port-forward.plist"
	@echo "Prometheus port-forward agent removed"
```
