## Problem

When the SSH tunnel goes down, the only recovery path is `bin/acg-refresh` run locally.
There is no Slack slash command to trigger a refresh remotely. The `/acg-status` output
already tells the user "run bin/acg-refresh" but that requires terminal access.

`/acg-refresh` should be a first-class Slack command alongside `/acg-up`, `/acg-down`,
and `/acg-status`.

**Exact new block:**

```python
def _run_cluster_refresh(job_id, response_url, thread_ts=None):
    """Run bin/acg-refresh --no-login-prompt; post result to response_url or thread."""
    job_dir = JOB_DIR / job_id
    lines = []

    def _log(msg):
        lines.append(msg + "\n")

    def _finish(status):
        (job_dir / "status").write_text(status)
        (job_dir / "output").write_text("".join(lines))
        text = "".join(lines)
        if thread_ts:
            _post_slack_bot(text, thread_ts=thread_ts)
        elif response_url:
            _slack_post(response_url, text)

    try:
        _log("🔄 *ACG Refresh* — refreshing credentials + SSH tunnel…")
        repo_root = pathlib.Path(__file__).resolve().parent.parent
        refresh_bin = repo_root / "bin" / "acg-refresh"
        result = subprocess.run(
            [str(refresh_bin), "--no-login-prompt"],
            capture_output=True, text=True, timeout=300,
        )
        if result.returncode == 0:
            _log("\n✅ *Refresh complete* — tunnel and credentials restored")
        else:
            stderr_tail = result.stderr.strip().splitlines()[-5:] if result.stderr.strip() else []
            _log("\n❌ *Refresh failed* (exit {})".format(result.returncode))
            if stderr_tail:
                _log("```\n{}\n```".format("\n".join(stderr_tail)))
        _finish("success")
    except subprocess.TimeoutExpired:
        _log("\n❌ *Refresh timed out* after 300s — run `bin/acg-refresh` manually")
        _finish("failed")
    except Exception as exc:
        _log(f"ERROR: {exc}")
        _finish("failed")


def _run_cluster_status(job_id, response_url, thread_ts=None):
```
