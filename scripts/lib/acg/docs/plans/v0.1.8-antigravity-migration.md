## 1. Context & Goal

Google retired the Node-based `gemini-cli` and replaced it with the Go-based **Antigravity CLI (`agy`)**. The original design had `_cdp_ensure_acg_session()` (in `scripts/lib/cdp.sh`) hand a natural-language prompt to `gemini-cli --yolo`, which then wrote and ran a Playwright script to verify the Pluralsight (ACG) login over CDP.

**This approach is dead and cannot be ported to `agy`:**

1. `_gemini_prompt` **no longer exists** anywhere in lib-acg or lib-foundation — the wrapper was removed in an earlier decouple pass, leaving `cdp.sh:121` calling an undefined function. `_cdp_ensure_acg_session()` is therefore currently broken (and, separately, **has no callers** — see §5).
2. A live smoke test (2026-06-21) proved **`agy` is not a drop-in for `gemini-cli`'s headless prompt executor**:
   - It ignores an inline imperative prompt and instead reads project context (`memory-bank/`, `GEMINI.md`, `AGENTS.md`, skills), then **refuses to act** without a formal task spec.
   - `--dangerously-skip-permissions` is a **no-op** in `agy` (it's a Claude Code flag); `agy` controls approvals via `~/.gemini/settings.json` `toolPermission`.
   - Output is verbose/non-deterministic — unsafe for code that greps a single `ACG_SESSION_OK` token + exit code.

**Decision (owner, 2026-06-21):** Drop the LLM agent from this path entirely. The session check is fully deterministic and needs no agent. Ship a static Playwright script that lib-acg runs directly with `node`. This removes the `gemini`/`agy` dependency from `_cdp_ensure_acg_session()` and aligns with the prior antigravity-decouple effort.

### Change 2.2 — NEW FILE `scripts/lib/acg_session_check.js`

Deterministic Playwright script. Connects to the **already-running** CDP browser (does not launch one, does not close it), checks the Pluralsight session, and — if logged out — prompts for manual login and polls until the sign-in URL clears. Prints exactly `ACG_SESSION_OK` on success; non-zero exit on failure/timeout.

**Exact new file content:**
```javascript
'use strict';
// scripts/lib/acg_session_check.js
// Verify the Pluralsight (ACG) session in the already-running CDP browser on :9222.
// Prints exactly "ACG_SESSION_OK" and exits 0 when a session is confirmed.
// Exits 1 (with an "ERROR:" line on stderr) when it cannot be confirmed.
//
// Env:
//   CDP_URL                 CDP endpoint (default http://localhost:9222)
//   ACG_LOGIN_TIMEOUT_MS    manual-login wait budget in ms (default 300000)

const { chromium } = require('playwright');

const CDP_URL = process.env.CDP_URL || 'http://localhost:9222';
const SANDBOX_URL = 'https://app.pluralsight.com/hands-on/playground/cloud-sandboxes';
const SIGNIN_URL = 'https://app.pluralsight.com/id/signin';
const LOGIN_TIMEOUT_MS = Number(process.env.ACG_LOGIN_TIMEOUT_MS || 300000);
const POLL_INTERVAL_MS = 5000;

async function isLoggedIn(page) {
  const selectors = [
    "[data-testid='user-menu']",
    "[aria-label='User menu']",
    "[data-testid='avatar']",
  ];
  for (const sel of selectors) {
    const visible = await page.locator(sel).first().isVisible().catch(() => false);
    if (visible) return true;
  }
  const heading = page.getByRole('heading', { name: /Cloud Sandboxes/i });
  return await heading.first().isVisible().catch(() => false);
}

(async () => {
  let browser;
  try {
    browser = await chromium.connectOverCDP(CDP_URL);
  } catch (err) {
    console.error('ERROR: cannot connect to CDP browser at ' + CDP_URL + ': ' + err.message);
    process.exit(1);
  }

  const contexts = browser.contexts();
  if (contexts.length === 0) {
    console.error('ERROR: no existing browser context on CDP endpoint');
    process.exit(1);
  }
  const context = contexts[0];
  const page = context.pages()[0] || (await context.newPage());

  try {
    await page.goto(SANDBOX_URL, { waitUntil: 'domcontentloaded', timeout: 60000 });
  } catch (err) {
    console.error('ERROR: navigation to sandbox page failed: ' + err.message);
    process.exit(1);
  }

  if (await isLoggedIn(page)) {
    console.log('ACG_SESSION_OK');
    process.exit(0);
  }

  await page.goto(SIGNIN_URL, { waitUntil: 'domcontentloaded', timeout: 60000 }).catch(() => {});
  console.log('ACTION REQUIRED: log into Pluralsight in the Antigravity browser window. Waiting up to '
    + Math.round(LOGIN_TIMEOUT_MS / 1000) + 's...');

  const deadline = Date.now() + LOGIN_TIMEOUT_MS;
  while (Date.now() < deadline) {
    await new Promise((r) => setTimeout(r, POLL_INTERVAL_MS));
    if (!page.url().includes('/signin')) {
      console.log('ACG_SESSION_OK');
      process.exit(0);
    }
  }

  console.error('ERROR: Pluralsight login timeout');
  process.exit(1);
})().catch((err) => {
  console.error('ERROR: unexpected failure: ' + (err && err.message ? err.message : String(err)));
  process.exit(1);
});
```

> **Do NOT call `browser.close()`** — the browser is the shared user Chrome on :9222; closing it over CDP would kill the user's session. Connecting and leaving it open is intentional.

### Change 2.3 — `scripts/lib/cdp.sh`: rewrite `_cdp_ensure_acg_session()` (lines 96–122) to run the static script

**Exact old block:**
```bash
function _cdp_ensure_acg_session() {
  if [[ "${K3DM_ACG_SKIP_SESSION_CHECK:-0}" == "1" ]]; then
    _info "K3DM_ACG_SKIP_SESSION_CHECK=1 — skipping ACG/Pluralsight session check"
    return 0
  fi
  _info "Checking Pluralsight (ACG) session in Gemini browser..."

  local gemini_prompt
  gemini_prompt="You are a browser automation agent. Use Playwright (Node.js) to do the following:

1. Connect to the running Gemini browser via CDP: const browser = await chromium.connectOverCDP('http://localhost:9222');
2. Use the first browser context and page (do NOT launch a new browser).
3. Navigate to https://app.pluralsight.com/hands-on/playground/cloud-sandboxes and wait for the page to load.
4. Check if the user is logged in by looking for user avatar, account menu, or sandbox list elements (e.g. [data-testid='user-menu'], [aria-label='User menu'], or a heading containing 'Cloud Sandboxes').
5. If logged in: print ACG_SESSION_OK and exit with code 0.
6. If NOT logged in:
   a. Navigate to https://app.pluralsight.com/id/signin
   b. Print: ACTION REQUIRED: Please log into Pluralsight in the Gemini browser window, then press Enter to continue.
   c. Wait for the page URL to no longer contain '/signin' — poll every 5 seconds, timeout after 300 seconds.
   d. Once URL is no longer '/signin', print ACG_SESSION_OK and exit with code 0.
   e. If 300 seconds pass without login, print ERROR: Pluralsight login timeout and exit with code 1.

Write the Playwright script to ${HOME}/.gemini/tmp/k3d-manager/ag_acg_session.js, execute with node, print the result.
Exit code 1 if session cannot be confirmed."

  _gemini_prompt "$gemini_prompt" --yolo
}
```

**Exact new block:**
```bash
function _cdp_ensure_acg_session() {
  if [[ "${K3DM_ACG_SKIP_SESSION_CHECK:-0}" == "1" ]]; then
    _info "K3DM_ACG_SKIP_SESSION_CHECK=1 — skipping ACG/Pluralsight session check"
    return 0
  fi
  _info "Checking Pluralsight (ACG) session via Playwright over CDP..."

  local _acg_root="${_LIB_ACG_ROOT:-$(cd "$(dirname "${BASH_SOURCE[0]}")/../.." && pwd)}"
  local _session_script="${_acg_root}/scripts/lib/acg_session_check.js"
  if [[ ! -f "${_session_script}" ]]; then
    _err "[acg] session-check script not found: ${_session_script}"
  fi
  if ! command -v node >/dev/null 2>&1; then
    _err "[acg] node is required for the ACG session check — install Node.js"
  fi

  NODE_PATH="${_acg_root}/node_modules" node "${_session_script}"
}
```

Notes:
- The contract is preserved: exit 0 when a session is confirmed, non-zero otherwise (node's exit code propagates).
- `NODE_PATH` resolves `playwright` from lib-acg's `node_modules`, matching the preflight pattern already used in `scripts/plugins/acg.sh` (around line 292).
